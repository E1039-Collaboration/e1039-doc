60 int SQChamberRealization::process_event(PHCompositeNode* topNode)
61 {
62  if (Verbosity() >= 2) cout << "SQChamberRealization: n_hits=" << m_hit_vec->size() << endl;
63  SQHitVector::Iter it = m_hit_vec->begin();
64  while (it != m_hit_vec->end()) {
65  SQHit* hit = *it;
66  int det_id = hit->get_detector_id();
67  PlaneParam* param = &list_param[det_id];
68  if (! param->on) {
69  it++;
70  continue;
71  }
72 
73  int ele_id = hit->get_element_id();
74  bool is_eff = (gRandom->Rndm() < param->eff);
75  if (Verbosity() >= 2) {
76  string det_name = GeomSvc::instance()->getDetectorName(det_id);
77  cout << " Hit: det=" << det_id << ":" << det_name << " ele=" << ele_id << " is_eff=" << is_eff << endl;
78  }
79  if (! is_eff) {
80  it = m_hit_vec->erase(it);
81  continue;
82  }
83 
84  CalibParamXT::Set* xt = m_cal_xt->GetParam(det_id);
85  if (xt) {
86  double dist = hit->get_drift_distance();
87  int dist_sign = dist > 0 ? +1 : -1;
88  dist *= dist_sign; // Made positive.
89 
90  double dx;
91  if (param->reso_fixed >= 0) dx = param->reso_fixed;
92  else dx = xt->x2dx.Eval(dist);
93  if (param->reso_scale >= 0) dx *= param->reso_scale;
94  if (Verbosity() >= 2) cout << " dist=" << dist << " dx=" << dx << " X0=" << xt->X0 << " X1=" << xt->X1 << endl;
95 
96  double dist_new;
97  int n_try = 10000;
98  while (n_try > 0) {
99  dist_new = gRandom->Gaus(dist, dx);
100  if (xt->X0 <= dist_new && dist_new <= xt->X1) break;
101  n_try--;
102  }
103  if (n_try == 0) {
104  cout << PHWHERE << " Failed at generating an in-range drift time. Something unexpected. Abort." << endl;
105  exit(1);
106  }
107  hit->set_drift_distance(dist_new * dist_sign);
108  hit->set_tdc_time(xt->x2t.Eval(dist_new));
109  }
110  it++;
111  }
112  return Fun4AllReturnCodes::EVENT_OK;
113 }
