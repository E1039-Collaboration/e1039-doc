77 void P6DExtDecayerPhysics::ConstructProcess()
78 {
81 
82  // Create Geant4 external decayer
83  G4Pythia6Decayer* extDecayer = new G4Pythia6Decayer();
84  extDecayer->SetVerboseLevel(0);
85 
86  aParticleIterator->reset();
87  int decayer_used = 0;
88  while ((*aParticleIterator)())
89  {
90  G4ParticleDefinition* particle = aParticleIterator->value();
91  G4ProcessManager* pmanager = particle->GetProcessManager();
92 
93  if ( verboseLevel > 1 ) {
94  G4cout << "Setting ext decayer for: "
95  << aParticleIterator->value()->GetParticleName()
96  << G4endl;
97  }
98 
99  G4ProcessVector* processVector = pmanager->GetProcessList();
100  for (G4int i=0; i<processVector->length(); i++) {
101 
102  G4Decay* decay = dynamic_cast<G4Decay*>((*processVector)[i]);
103  if ( decay )
104  {
105  // The extDecayer will be deleted in G4Decay destructor
106  // increment counter in case we want to print out stats
107  // for whatever reason (non null means it is used and
108  // must not be deleted)
109  decay->SetExtDecayer(extDecayer);
110  decayer_used++;
111  }
112  }
113  }
114 
115  if (_active_force_decay) {
116  extDecayer->ForceDecayType(_force_decay_type);
117  }
118 
119  // If the extDecayer isn't used for this particle we need to delete it here
120  // as far as I see this never happens but this makes coverity happy
121  if (! decayer_used)
122  {
123  delete extDecayer;
124  }
125  if ( verboseLevel > 0 ) {
126  G4cout << "External decayer physics constructed." << G4endl;
127  }
128 }
