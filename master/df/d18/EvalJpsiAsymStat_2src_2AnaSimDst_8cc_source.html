258 void AnaSimDst::CalcAngle(const TLorentzVector& mu0, const TLorentzVector& mu1, const int pol_sign, double& xb, double& xt, double& phi_s_tf)
259 {
260  //Initial 4-vector definition
261  const float pro_mass = 0.93827;
262  const float beamE = 120.0;
263  TLorentzVector hadron_b, hadron_t;
264  hadron_b.SetPxPyPzE(0.0,0.0,TMath::Sqrt(beamE*beamE-pro_mass*pro_mass),beamE);
265  hadron_t.SetPxPyPzE(0.0,0.0,0.0,pro_mass);
266  TLorentzVector initial = hadron_b + hadron_t;
267 
268  TLorentzVector spin;
269  spin.SetPxPyPzE(0.0, (pol_sign > 0 ? +1.0 : -1.0), 0.0, 0.0);
270 
271  TLorentzVector dimu = mu0 + mu1;
272  float dimu_gmass = dimu.Mag();
273 
274  //boost to centre of mass
275  TLorentzRotation cm_r;
276  TVector3 cm_v = initial.BoostVector();
277  cm_r.Boost(-cm_v);
278 
279  TLorentzVector hadron_b_cm = cm_r * hadron_b;
280  TLorentzVector hadron_t_cm = cm_r * hadron_t;
281  TLorentzVector mu0_cm = cm_r * mu0;
282  TLorentzVector mu1_cm = cm_r * mu1;
283  TLorentzVector dimu_cm = cm_r * dimu;
284  TLorentzVector spin_cm = cm_r * spin;
285 
286  float s = 2.0*beamE*pro_mass + 2.0*pro_mass*pro_mass;
287  float sqrt_s = TMath::Sqrt(s);
288  float dimu_E = dimu_cm.E();
289  float dimu_pl = dimu_cm.Pz();
290  float dimu_pl_max = sqrt_s/2.0*(1-(dimu_gmass*dimu_gmass)/s);
291  float xf = dimu_pl/dimu_pl_max;
292  float y = 0.5*TMath::Log((dimu_E+dimu_pl)/(dimu_E-dimu_pl));
293  float tau = (dimu_gmass*dimu_gmass)/s;
294  xb = 0.5*(TMath::Sqrt(xf*xf+4*tau)+xf);
295  xt = 0.5*(TMath::Sqrt(xf*xf+4*tau)-xf);
296 
297  //boost to dilepton restframe
298  TLorentzRotation cm_dimu;
299  TVector3 cm_v2 = dimu_cm.BoostVector();
300  cm_dimu.Boost(-cm_v2);
301 
302  TLorentzVector hadron_b_cm_dimu = cm_dimu * hadron_b_cm;
303  TLorentzVector hadron_t_cm_dimu = cm_dimu * hadron_t_cm;
304  TLorentzVector mu0_cm_dimu = cm_dimu * mu0_cm;
305  TLorentzVector mu1_cm_dimu = cm_dimu * mu1_cm;
306  TLorentzVector dimu_cm_dimu = cm_dimu * dimu_cm;
307  TLorentzVector spin_cm_dimu = cm_dimu * spin_cm;
308 
309  TVector3 zaxis = (hadron_b_cm_dimu.Vect()).Unit()-(hadron_t_cm_dimu.Vect()).Unit();
310  TVector3 zunit = zaxis.Unit();
311  TVector3 hadron_plane_norm = zunit.Cross((hadron_b_cm_dimu.Vect()).Unit());
312  TVector3 lepton_plane_norm = zunit.Cross((mu0_cm_dimu.Vect()).Unit());
313  TVector3 yunit = lepton_plane_norm.Unit();
314  TVector3 xunit = (yunit.Cross(zunit)).Unit();
315 
316  //below are phi_s_tf from target rest frame. The frame for the Asymmetry calculation
317  TVector3 spin_vect_tf = spin.Vect();
318  TVector3 z_prime_unit = (hadron_b.Vect()).Unit();
319  TVector3 y_prime_unit = (z_prime_unit.Cross(dimu.Vect())).Unit();
320  TVector3 x_prime_unit = (y_prime_unit.Cross(z_prime_unit)).Unit();
321  float cosphi_s_tf = (x_prime_unit.Dot(spin_vect_tf))/((x_prime_unit.Mag())*(spin_vect_tf.Mag()));
322  //float phi_s_tf;
323  if ( acos( spin_vect_tf.Dot(y_prime_unit) / ( (spin_vect_tf).Mag() * y_prime_unit.Mag() ) ) > TMath::Pi()/2 )
324  {
325  phi_s_tf = -acos(cosphi_s_tf);
326  }
327  else
328  {
329  phi_s_tf = acos(cosphi_s_tf);
330  }
331 }
