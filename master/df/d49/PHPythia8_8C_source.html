172 int PHPythia8::process_event(PHCompositeNode *topNode)
173 {
174  if (verbosity >= VERBOSITY_MORE) cout << "PHPythia8::process_event - event: " << _eventcount << endl;
175 
176  bool passedGen = false;
177  bool passedTrigger = false;
178  int genCounter = 0;
179 
180  while (!passedTrigger)
181  {
182  ++genCounter;
183 
184  // generate another pythia event
185  while (!passedGen)
186  {
187  passedGen = _pythia->next();
188  }
189 
190  // test trigger logic
191 
192  bool andScoreKeeper = true;
193  if (verbosity >= VERBOSITY_EVEN_MORE)
194  {
195  cout << "PHPythia8::process_event - triggersize: " << _registeredTriggers.size() << endl;
196  }
197 
198  for (unsigned int tr = 0; tr < _registeredTriggers.size(); tr++)
199  {
200  bool trigResult = _registeredTriggers[tr]->Apply(_pythia);
201 
202  if (verbosity >= VERBOSITY_EVEN_MORE)
203  {
204  cout << "PHPythia8::process_event trigger: "
205  << _registeredTriggers[tr]->GetName() << " " << trigResult << endl;
206  }
207 
208  if (_triggersOR && trigResult)
209  {
210  passedTrigger = true;
211  break;
212  }
213  else if (_triggersAND)
214  {
215  andScoreKeeper &= trigResult;
216  }
217 
218  if (verbosity >= VERBOSITY_EVEN_MORE && !passedTrigger)
219  {
220  cout << "PHPythia8::process_event - failed trigger: "
221  << _registeredTriggers[tr]->GetName() << endl;
222  }
223  }
224 
225  if ((andScoreKeeper && _triggersAND) || (_registeredTriggers.size() == 0))
226  {
227  passedTrigger = true;
228  genCounter = 0;
229  }
230 
231  passedGen = false;
232  }
233 
234  // fill HepMC object with event & pass to
235 
236  HepMC::GenEvent *genevent = new HepMC::GenEvent(HepMC::Units::GEV, HepMC::Units::MM);
237  _pythiaToHepMC->fill_next_event(*_pythia, genevent, _eventcount);
238 
239  /* pass HepMC to PHNode*/
240  PHHepMCGenEvent *success = hepmc_helper.insert_event(genevent);
241  if (!success)
242  {
243  cout << "PHPythia8::process_event - Failed to add event to HepMC record!" << endl;
244  return Fun4AllReturnCodes::ABORTRUN;
245  }
246 
247  // print outs
248 
249  if (verbosity >= VERBOSITY_MORE) cout << "PHPythia8::process_event - FINISHED WHOLE EVENT" << endl;
250  if (_eventcount < 2 && verbosity >= VERBOSITY_SOME) _pythia->event.list();
251  if (_eventcount >= 2 && verbosity >= VERBOSITY_A_LOT) _pythia->event.list();
252 
253  ++_eventcount;
254 
255  // save statistics
256  if (_integral_node)
257  {
258  _integral_node->set_N_Generator_Accepted_Event(_pythia->info.nAccepted());
259  _integral_node->set_N_Processed_Event(_eventcount);
260  _integral_node->set_Sum_Of_Weight(_pythia->info.weightSum());
261  _integral_node->set_Integrated_Lumi(_pythia->info.nAccepted() / (_pythia->info.sigmaGen() * 1e9));
262  }
263 
264  return Fun4AllReturnCodes::EVENT_OK;
265 }
