238 bool DoEmbedding::GetNextEmbEvent()
239 {
240  if (! m_file_emb) { // Need to open an embedding-data file
241  if (m_idx_emb_file >= m_list_emb_file.size()) return false; // No file available.
242  string fn_root = m_list_emb_file[m_idx_emb_file];
243  OpenEmbDataFile(fn_root.c_str());
244  m_tree_emb->SetBranchAddress("emb_evt" , &m_emb_evt );
245  m_tree_emb->SetBranchAddress("emb_hit_list", &m_emb_hit_list);
246  m_emb_data_has_sim_evt = (m_tree_emb->FindBranch("emb_sim_evt" ) != 0);
247  m_emb_data_has_sim_trk = (m_tree_emb->FindBranch("emb_sim_trk_list") != 0);
248  m_emb_data_has_sim_dim = (m_tree_emb->FindBranch("emb_sim_dim_list") != 0);
249  if (m_emb_data_has_sim_evt) m_tree_emb->SetBranchAddress("emb_sim_evt" , &m_emb_sim_evt );
250  if (m_emb_data_has_sim_trk) m_tree_emb->SetBranchAddress("emb_sim_trk_list", &m_emb_sim_trk_list);
251  if (m_emb_data_has_sim_dim) m_tree_emb->SetBranchAddress("emb_sim_dim_list", &m_emb_sim_dim_list);
252 
253  m_idx_emb_file++;
254  m_idx_emb_evt = 0;
255  }
256 
257  if (m_idx_emb_evt >= m_tree_emb->GetEntries()) { // No event available.
258  CloseEmbDataFile();
259  return GetNextEmbEvent();
260  }
261 
262  m_tree_emb->GetEntry(m_idx_emb_evt++);
263 
264  return true;
265 }
