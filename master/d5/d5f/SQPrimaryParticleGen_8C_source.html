255 //=====================DrellYan=====================================
257 double SQPrimaryParticleGen::CrossSectionDrellYan(const double mass, const double xF, const double x1, const double x2, const double pARatio)
258 {
259  double zOverA = pARatio;
260  double nOverA = 1. - zOverA;
261 
262  double dbar1 = _pdf->xfxQ(-1, x1, mass)/x1;
263  double ubar1 = _pdf->xfxQ(-2, x1, mass)/x1;
264  double d1 = _pdf->xfxQ( 1, x1, mass)/x1;
265  double u1 = _pdf->xfxQ( 2, x1, mass)/x1;
266  double s1 = _pdf->xfxQ( 3, x1, mass)/x1;
267  double c1 = _pdf->xfxQ( 4, x1, mass)/x1;
268 
269  double dbar2 = _pdf->xfxQ(-1, x2, mass)/x2;
270  double ubar2 = _pdf->xfxQ(-2, x2, mass)/x2;
271  double d2 = _pdf->xfxQ( 1, x2, mass)/x2;
272  double u2 = _pdf->xfxQ( 2, x2, mass)/x2;
273  double s2 = _pdf->xfxQ( 3, x2, mass)/x2;
274  double c2 = _pdf->xfxQ( 4, x2, mass)/x2;
275 
276  double xsec_pdf = 4./9.*(u1*(zOverA*ubar2 + nOverA*dbar2) + ubar1*(zOverA*u2 + nOverA*d2) + 2*c1*c2) +
277  1./9.*(d1*(zOverA*dbar2 + nOverA*ubar2) + dbar1*(zOverA*d2 + nOverA*u2) + 2*s1*s2);
278 
279  // KFactor
280  double xsec_kfactor;
281  if (mass < 2.5) xsec_kfactor = 1.25;
282  else if (mass < 7.5) xsec_kfactor = 1.25 + (1.82 - 1.25)*(mass - 2.5)/5.;
283  else xsec_kfactor = 1.82;
284 
285  //phase space
286  double mass_MeV = mass * (CLHEP::GeV/CLHEP::MeV);
287  double xsec_phsp = x1*x2/(x1 + x2)/pow(mass_MeV, 3);
288 
289  //generation limitation
290  double xsec_limit = (pow(cosThetaMax, 3)/3 + cosThetaMax - pow(cosThetaMin, 3)/3 - cosThetaMin) * 3/8;
291 
292  // hbarc_squared is in MeV*mm^2, given by Geant4.
293  const double xsec_const = 8.0 / 9.0 * pi * pow(CLHEP::fine_structure_const, 2) * CLHEP::hbarc_squared;
294 
295  // Cross section in pb/GeV
296  return xsec_pdf * xsec_kfactor * xsec_phsp * xsec_limit * xsec_const * (CLHEP::mm2/CLHEP::picobarn) / (CLHEP::MeV/CLHEP::GeV);
297 }

322 double SQPrimaryParticleGen::CrossSectionJPsi(const double xF)
323 {
324  const double A = 1.464e6*TMath::Exp(-16.66*DPGEN::mjpsi/DPGEN::sqrts); // Taken from Tab. 1 of PRD52_1307
325  const double W = 0.2398; // Jpsi xf gaussian width. Taken from where?
326  const double BR = 0.0594; // Branching ratio of J/psi -> mumu. Seen above Fig. 4 of PRD52_1307.
327  return BR * A * TMath::Exp(-xF*xF/W/W/2) / (DPGEN::sqrt2pi * W);
328 }

342 int SQPrimaryParticleGen::generateDrellYan(const TVector3& vtx, const double pARatio, double luminosity)
343 {
344  //sets invaraint mass and xF = x1-x2 for virtual photon
345  double mass = gRandom->Uniform(0,1)*(massMax - massMin) + massMin;
346  double xF = gRandom->Uniform(0,1)*(xfMax - xfMin) + xfMin;
347 
348  if(!generateDimuon(mass, xF)) return 1;
349 
350  InsertMuonPair(vtx);
351 
352  double dim_mass, dim_pT, dim_x1, dim_x2, dim_xF;
353  UtilDimuon::CalcVar(_dim_gen, dim_mass, dim_pT, dim_x1, dim_x2, dim_xF);
354 
355  double xsec = CrossSectionDrellYan(dim_mass, dim_xF, dim_x1, dim_x2, pARatio);
356  double weight = xsec * luminosity * (massMax - massMin) * (xfMax - xfMin);
357  InsertEventInfo(xsec, weight, vtx);
358 
359  return 0;
360 }

457 bool SQPrimaryParticleGen::generateDimuon(double mass, double xF)
458 {
459  double pz = xF*(DPGEN::sqrts - mass*mass/DPGEN::sqrts)/2.;
460 
461  //double pTmaxSq = (DPGEN::s*DPGEN::s*(1. - xF*xF) - 2.*DPGEN::s*mass*mass + mass*mass*mass*mass)/DPGEN::s/4.;
462  double pTmaxSq = (DPGEN::s / 4) * (1 - mass*mass/DPGEN::s)*(1 - mass*mass/DPGEN::s) * (1 - xF*xF);
463  if(pTmaxSq < 0.)
464  {
465  cout << PHWHERE << "pTmaxSq < 0." << endl;
466  return false;
467  }
468 
469  double pTmax = sqrt(pTmaxSq);
470  double pT = 10.;
471 
472  if(pTmax < 0.3)
473  {
474  pT = pTmax*sqrt(gRandom->Uniform(0,1));
475  }
476  else if(_DrellYanGen) // PRD43, 2815 (1991)
477  {
478  while(pT > pTmax) pT = _pT0DY*TMath::Sqrt(1./TMath::Power(gRandom->Uniform(0,1), _pTpowDY) - 1.);
479  }
480  else
481  {
482  while(pT > pTmax) pT = _pT0JPsi*TMath::Sqrt(1./TMath::Power(gRandom->Uniform(0,1), _pTpowJPsi) - 1.);
483  }
484 
485  double phi = gRandom->Uniform(0,1)*DPGEN::twopi;
486  double px = pT*TMath::Cos(phi);
487  double py = pT*TMath::Sin(phi);
488 
489  //configure phase space generator
490  TLorentzVector p_dimuon;
491  p_dimuon.SetXYZM(px, py, pz, mass);
492  p_dimuon.Boost(DPGEN::bv_cms);
493  double masses[2] = {0.105658,0.105658};// Mass of muons
494  phaseGen.SetDecay(p_dimuon, 2, masses);
495 
496  const int N_TRY = 1000000;
497  for (int i_try = 0; i_try < N_TRY; i_try++) // Loop to generate a proper kinematics.
498  {
499  phaseGen.Generate();
500  _dim_gen->set_mom_pos(*(phaseGen.GetDecay(0)));
501  _dim_gen->set_mom_neg(*(phaseGen.GetDecay(1)));
502  double dim_mass, dim_pT, dim_x1, dim_x2, dim_xF;
503  UtilDimuon::CalcVar(_dim_gen, dim_mass, dim_pT, dim_x1, dim_x2, dim_xF);
504  if(dim_x1 < x1Min || dim_x1 > x1Max) continue;
505  if(dim_x2 < x2Min || dim_x2 > x2Max) continue;
506  double dim_costh, dim_phi;
507  UtilDimuon::Lab2CollinsSoper(_dim_gen, dim_costh, dim_phi);
508  if(dim_costh < cosThetaMin || dim_costh >cosThetaMax) continue;
509  if(_DrellYanGen && gRandom->Uniform(0,2) > 1. + pow(dim_costh, 2)) continue;
510  return true; // A proper dimuon is generated.
511  }
512  cout << PHWHERE << "No proper dimuon was generated. Is your kinematic range reasonable?" << endl;
513  return false;
514 }
