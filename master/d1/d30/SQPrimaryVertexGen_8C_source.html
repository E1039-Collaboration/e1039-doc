31 void MaterialProfile::calcProb()
32 {
33  //sort the interactable pieces first
34  sort(interactables.begin(), interactables.end());
35 
36  //set the quanties that rely on its neighbours
37  double attenuationSum = 0.;
38  for(unsigned int i = 0; i < interactables.size(); ++i)
39  {
40  interactables[i].attenuationSelf = 1. - TMath::Exp(-interactables[i].length/interactables[i].nucIntLen);
41  interactables[i].attenuation = (1. - attenuationSum)*interactables[i].attenuationSelf;
42  interactables[i].prob = interactables[i].attenuation*interactables[i].density*interactables[i].nucIntLen;
43 
44  attenuationSum += interactables[i].attenuation;
45  }
46 
47  //set the accumulatedProbs
48  nPieces = interactables.size();
49 
50  interactables[0].accumulatedProb = 0.;
51  accumulatedProbs[0] = 0.;
52  for(unsigned int i = 1; i < nPieces; ++i)
53  {
54  interactables[i].accumulatedProb = interactables[i-1].accumulatedProb + interactables[i-1].prob;
55  accumulatedProbs[i] = interactables[i].accumulatedProb;
56  }
57  accumulatedProbs[nPieces] = accumulatedProbs[nPieces-1] + interactables.back().prob;
58  probSum = accumulatedProbs[nPieces];
59 
60  //Normalize the accumulated probability
61  for(unsigned int i = 0; i < nPieces+1; ++i)
62  {
63  accumulatedProbs[i] = accumulatedProbs[i]/accumulatedProbs[nPieces];
64  }
65 }

168 {
169  n_b = 0;

172  z_b[0] = z_curr;

246 TVector3 SQPrimaryVertexGen::generateVertex()
247 {
248  if(!inited) init();
249 
250  double xvtx, yvtx, zvtx;
251  generateVtxPerp(xvtx, yvtx);
252  bool inTarget = ((xvtx-targetX0)*(xvtx-targetX0)/targetSX/targetSX + (yvtx-targetY0)*(yvtx-targetX0)/targetSY/targetSY) < 1.;
253 
254  //keep generating x/y until it falls in target acceptance in targetOnlyMode
255  while(targetOnlyMode && (!inTarget))
256  {
257  generateVtxPerp(xvtx, yvtx);
258  inTarget = ((xvtx-targetX0)*(xvtx-targetX0)/targetSX/targetSX + (yvtx-targetY0)*(yvtx-targetX0)/targetSY/targetSY) < 1.;
259  }
260 
261  // if the x/y is within target acceptance (should happen majority of the time), use the pre-cached
262  // material profile to generate z vtx; otherwise use geometry navigator to find the material profile /
263  // on the path of this proton
264  MaterialProfile* activeProf;
265  if(!inTarget)
266  {
267  activeProf = new MaterialProfile;
268  fillMaterialProfile(activeProf, xvtx, yvtx);
269  }
270  else
271  {
272  activeProf = defaultMatProf;
273  }
274 
275  int index = activeProf->findInteractingPiece(rndm.Rndm());
276  zvtx = activeProf->interactables[index].getZ(rndm.Rndm());
277  pARatio = activeProf->interactables[index].protonPerc();
278  probSum = activeProf->probSum;
279 
280  if(!inTarget) delete activeProf;
281  return TVector3(xvtx, yvtx, zvtx);

290 double SQPrimaryVertexGen::funcBeamProfile(double* val, double* par)
291 {
292  double x = val[0];
293  double y = val[1];
294  double x_c = par[0];
295  double x_s = par[1];
296  double y_c = par[2];
297  double y_s = par[3];
298 
299  const double r_pipe = 5.0; // in cm
300  const double r_boun = 3.0; // normalized boundary, where gaus changes to 1/r
301  if(x*x + y*y > r_pipe*r_pipe) { // not "x-x_c" nor "y-y_c" here
302  return 0.0;
303  }
304 
305  double x_nr = (x - x_c)/x_s; // normalized relative x
306  double y_nr = (y - y_c)/y_s; // normalized relative y
307  double r2_nr = x_nr*x_nr + y_nr*y_nr;
308  double r_nr = sqrt(r2_nr);
309  if(r_nr < r_boun) {
310  return exp(-0.5*r2_nr);
311  } else { // >= r_boun
312  return exp(-0.5*r_boun*r_boun)*r_boun/r_nr;
313  }
314 }
