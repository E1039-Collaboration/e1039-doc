78 int TestSimAnalyzer::process_event(PHCompositeNode* topNode) {
79 
80  if(Verbosity() >= Fun4AllBase::VERBOSITY_SOME)
81  std::cout << "Entering TestSimAnalyzer::process_event: " << _event << std::endl;
82 
83  ResetEvalVars();
84 
85  if(_spill_map) {
86  auto spill_info = _spill_map->get(_b_spill_id);
87  if(spill_info) {
88  _b_target_pos = spill_info->get_target_pos();
89  } else {
90  LogWarning("");
91  }
92  }
93 
94  if(_event_header) {
95  _b_event_id = _event_header->get_event_id();
96  _b_spill_id = _event_header->get_spill_id();
97  _b_run_id = _event_header->get_run_id();
98  }
99 
100  if(_hit_vector) {
101  _b_n_hits = 0;
102  for(auto iter = _hit_vector->begin(); iter!= _hit_vector->end();++iter) {
103  SQHit *hit = (*iter);
104  _b_hit_id[_b_n_hits] = (*iter)->get_hit_id();
105  _b_drift_distance[_b_n_hits] = (*iter)->get_drift_distance();
106  _b_pos[_b_n_hits] = (*iter)->get_pos();
107  _b_detector_z[_b_n_hits] = p_geomSvc->getPlanePosition((*iter)->get_detector_id());
108  _b_detector_id[_b_n_hits] = (*iter)->get_detector_id();
109  if(_truth) {
110  //int track_id = (*iter)->get_track_id();
111  //LogDebug(track_id);
112  _b_truth_x[_b_n_hits] = (*iter)->get_truth_x();
113  _b_truth_y[_b_n_hits] = (*iter)->get_truth_y();
114  _b_truth_z[_b_n_hits] = (*iter)->get_truth_z();
115 
116  double uVec[3] = {
117  p_geomSvc->getCostheta(hit->get_detector_id()),
118  p_geomSvc->getSintheta(hit->get_detector_id()),
119  0
120  };
121  _b_truth_pos[_b_n_hits] =
122  _b_truth_x[_b_n_hits]*uVec[0] + _b_truth_y[_b_n_hits]*uVec[1];
123 
124  //LogDebug("detector_id: " << _b_detector_id[_b_n_hits]);
125  }
126  ++_b_n_hits;
127  }
128  }
129 
130  //LogDebug("nhits: " << _hit_vector->size());
131 
132  _tout->Fill();
133 
134  if(Verbosity() >= Fun4AllBase::VERBOSITY_SOME)
135  std::cout << "Leaving TestSimAnalyzer::process_event: " << _event << std::endl;
136  ++_event;
137 
138  return Fun4AllReturnCodes::EVENT_OK;
139 }

151 int TestSimAnalyzer::InitEvalTree() {
152  PHTFileServer::get().open(_out_name.c_str(), "RECREATE");
153 
154  _tout = new TTree("T", "TestSimAnalyzer");
155  _tout->Branch("runID", &_b_run_id, "runID/I");
156  _tout->Branch("spillID", &_b_spill_id, "spillID/I");
157  _tout->Branch("liveProton", &_b_target_pos, "liveProton/F");
158  _tout->Branch("eventID", &_b_event_id, "eventID/I");
159  _tout->Branch("nHits", &_b_n_hits, "nHits/I");
160  _tout->Branch("hitID", _b_hit_id, "hitID[nHits]/I");
161  _tout->Branch("detectorID", _b_detector_id, "detectorID[nHits]/I");
162  _tout->Branch("detectorZ", _b_detector_z, "detectorZ[nHits]/F");
163  _tout->Branch("truth_x", _b_truth_x, "truth_x[nHits]/F");
164  _tout->Branch("truth_y", _b_truth_y, "truth_y[nHits]/F");
165  _tout->Branch("truth_z", _b_truth_z, "truth_z[nHits]/F");
166  _tout->Branch("truth_pos", _b_truth_pos, "truth_pos[nHits]/F");
167  _tout->Branch("driftDistance", _b_drift_distance, "driftDistance[nHits]/F");
168  _tout->Branch("pos", _b_pos, "pos[nHits]/F");
169 
170  return 0;
171 }
