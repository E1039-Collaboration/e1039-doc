156 double GFTrack::updatePropState(const TVectorD& meas, const TMatrixDSym& V)
157 {
158  //get the H matrix from measurement
159  std::unique_ptr<const genfit::AbsHMatrix> H(_virtMeas->constructHMatrix(_propState->getRep()));
160 
161  //Handcrafted KF
162  TVectorD stateVec(_propState->getState());
163  TMatrixDSym cov(_propState->getCov());
164  double chi2 = 9999.;
165  double ndf = meas.GetNrows();
166 
167  TVectorD res(meas - H->Hv(stateVec));
168  TMatrixDSym covSumInv(cov);
169  H->HMHt(covSumInv);
170  covSumInv += V;
171  genfit::tools::invertMatrix(covSumInv);
172 
173  TMatrixD CHt(H->MHt(cov));
174  TVectorD update(TMatrixD(CHt, TMatrixD::kMult, covSumInv)*res);
175 
176  stateVec += update;
177  covSumInv.Similarity(CHt);
178  cov -= covSumInv;
179  _propState->setStateCov(stateVec, cov);
180 
181  //calc chi2
182  TVectorD resNew(meas - H->Hv(stateVec));
183  TMatrixDSym HCHt(cov);
184  H->HMHt(HCHt);
185  HCHt -= V;
186  HCHt *= -1;
187  genfit::tools::invertMatrix(HCHt);
188  chi2 = HCHt.Similarity(resNew);
189 
190  return chi2/ndf;
191 }
