121 int OnlMonHodo::ProcessEventOnlMon(PHCompositeNode* topNode)
122 {
123  SQEvent* event_header = findNode::getClass<SQEvent >(topNode, "SQEvent");
124  SQHitVector* hit_vec = findNode::getClass<SQHitVector>(topNode, "SQHitVector");
125  if (!event_header || !hit_vec) return Fun4AllReturnCodes::ABORTEVENT;
126 
127  for (SQHitVector::ConstIter it = hit_vec->begin(); it != hit_vec->end(); it++) {
128  int det_id = (*it)->get_detector_id();
129  for (int i_det = 0; i_det < N_DET; i_det++) {
130  if (list_det_id[i_det] != det_id) continue;
131  int ele = (*it)->get_element_id();
132  double time = (*it)->get_tdc_time ();
133  h1_ele [i_det]->Fill(ele );
134  h1_time [i_det]->Fill(time);
135  h2_time_ele[i_det]->Fill(ele, time);
136  if ((*it)->is_in_time()) {
137  h1_ele_in [i_det]->Fill(ele );
138  h1_time_in[i_det]->Fill(time);
139  }
140  }
141  }
142 
143  return Fun4AllReturnCodes::EVENT_OK;
