97  ret = InitGeom(topNode);
98  if(ret != Fun4AllReturnCodes::EVENT_OK) return ret;
99 
100  PHField* field = PHFieldUtility::GetFieldMapNode(nullptr, topNode);
101  assert(field);
102 
103  if(verbosity > 2) {
104  cout << "PHField check: " << "-------" << endl;
105  std::ofstream field_scan("field_scan.csv");
106  field->identify(field_scan);
107  field_scan.close();
108  }
109 
111  //fastfinder = new KalmanDSTrk(field, _t_geo_manager, _enable_KF, _DS_level,_sim_db_name,_pattern_db_name);
112  fastfinder = new KalmanFastTracking(field, _t_geo_manager, _enable_KF);
113  fastfinder->Verbosity(verbosity);
114 
115  if(_enable_event_reducer) {
116  TString opt = "aoc"; //turn on after pulse removal, out of time removal, and cluster removal
117  if(p_jobOptsSvc->m_enableTriggerMask) opt = opt + "t";
118  if(p_jobOptsSvc->m_sagittaReducer) opt = opt + "s";
119  if(p_jobOptsSvc->m_updateAlignment) opt = opt + "e";

356  fastfinder->printTimers();

358  _recEvent->setRawEvent(_rawEvent);
359

361 
362  for(std::list<Tracklet>::iterator iter = rec_tracklets.begin(); iter != rec_tracklets.end(); ++iter)
363  {
364  iter->calcChisq();
365 #ifdef _DEBUG_ON
366  iter->print();
367 #endif
368 
369 #ifndef _ENABLE_KF
370  SRecTrack recTrack = iter->getSRecTrack();
371 #ifdef _DEBUG_ON

373 #endif

375 #endif
376  }
377 
378 #ifdef _ENABLE_KF
379  std::list<SRecTrack>& rec_tracks = fastfinder->getSRecTracks();
380  for(std::list<SRecTrack>::iterator iter = rec_tracks.begin(); iter != rec_tracks.end(); ++iter)
381  {
382 #ifdef _DEBUG_ON
383  iter->print();
384 #endif
385  _recEvent->insertTrack(*iter);
386  }
387 #endif
388 
389  _tout->Fill();
390 
391  if(Verbosity() >= Fun4AllBase::VERBOSITY_SOME)
392  std::cout << "Leaving KalmanFastTrackingWrapper::process_event: " << _event << std::endl;
393  ++_event;
394 
395  return Fun4AllReturnCodes::EVENT_OK;
396 }
