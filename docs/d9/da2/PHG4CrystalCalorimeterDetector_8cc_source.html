310 
311  /* Open the datafile, if it won't open return an error */
312  ifstream istream_mapping;
313  if (!istream_mapping.is_open())
314  {
315  istream_mapping.open( _mapping_tower_file.c_str() );
316  if(!istream_mapping)

319  exit(1);
320  }
321  }
322 
323  /* loop over lines in file */
324  string line_mapping;
325  while ( getline( istream_mapping, line_mapping ) )
326  {
327  /* Skip lines starting with / including a '#' */
328  if ( line_mapping.find("#") != string::npos )
329  {
330  if ( verbosity > 0 )
331  {
332  cout << "PHG4CrystalCalorimeterDetector: SKIPPING line in mapping file: " << line_mapping << endl;
333  }
334  continue;
335  }
336 
337  istringstream iss(line_mapping);
338 
339  /* If line starts with keyword Tower, add to tower positions */
340  if ( line_mapping.find("Tower ") != string::npos )
341  {
342  unsigned idx_j, idx_k, idx_l;
343  G4double pos_x, pos_y, pos_z;
344  G4double size_x, size_y, size_z;
345  G4double rot_x, rot_y, rot_z;
346  G4double dummy;
347  string dummys;
348 
349  /* read string- break if error */
350  if ( !( iss >> dummys >> dummy >> idx_j >> idx_k >> idx_l >> pos_x >> pos_y >> pos_z >> size_x >> size_y >> size_z >> rot_x >> rot_y >> rot_z ) )
351  {
352  cerr << "ERROR in PHG4CrystalCalorimeterDetector: Failed to read line in mapping file " << _mapping_tower_file << endl;
353  exit(1);
354  }
355 
356  /* Construct unique name for tower */
357  /* Mapping file uses cm, this class uses mm for length */
358  ostringstream towername;
359  towername.str("");
360  towername << _towerlogicnameprefix << "_j_" << idx_j << "_k_" << idx_k;
361 
362  /* Add Geant4 units */
363  pos_x = pos_x * cm;
364  pos_y = pos_y * cm;
365  pos_z = pos_z * cm;
366 
367  /* insert tower into tower map */
368  towerposition tower_new;
369  tower_new.x = pos_x;
370  tower_new.y = pos_y;
371  tower_new.z = pos_z;
372  _map_tower.insert( make_pair( towername.str() , tower_new ) );
373 
374  }
375  else
376  {
377  /* If this line is not a comment and not a tower, save parameter as string / value. */
378  string parname;
379  G4double parval;
380 
381  /* read string- break if error */
382  if ( !( iss >> parname >> parval ) )
383  {
384  cerr << "ERROR in PHG4CrystalCalorimeterDetector: Failed to read line in mapping file " << _mapping_tower_file << endl;
385  exit(1);
386  }
387 
388  _map_global_parameter.insert( make_pair( parname , parval ) );
389 
390  }
391  }

394  std::map<string,G4double>::iterator parit;
395

403
