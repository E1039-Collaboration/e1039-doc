61 bool PHG4mRICHSteppingAction::UserSteppingAction( const G4Step* aStep, bool )
62 {
63  G4TouchableHandle touch = aStep->GetPreStepPoint()->GetTouchableHandle();
64  G4VPhysicalVolume* volume = touch->GetVolume();
65 
66  // detector_->IsInForwardEcal(volume)
67  // returns
68  // 0 is outside of Forward ECAL
69  // 1 is inside scintillator
70  // -1 is inside absorber (dead material)
71 
72  int whichactive = detector_->IsInmRICH(volume);
73 
74  if ( !whichactive ) return false;
75 
76  /* Get energy deposited by this step */
77  G4double edep = aStep->GetTotalEnergyDeposit() / GeV;
78  G4double eion = (aStep->GetTotalEnergyDeposit() - aStep->GetNonIonizingEnergyDeposit()) / GeV;
79 
80  /* Get pointer to associated Geant4 track */
81  const G4Track* aTrack = aStep->GetTrack();
82 
83  // if this block stops everything, just put all kinetic energy into edep
84  if (IsBlackHole) {
85  edep = aTrack->GetKineticEnergy() / GeV;
86  G4Track* killtrack = const_cast<G4Track *> (aTrack);
87  killtrack->SetTrackStatus(fStopAndKill);
88  }
89 
90  /* Make sure we are in a volume */
91  if ( active ) {
92  /* Check if particle is 'geantino' */
93  bool geantino = false;
94  if (aTrack->GetParticleDefinition()->GetPDGEncoding() == 0 &&
95  aTrack->GetParticleDefinition()->GetParticleName().find("geantino") != string::npos) {
96  geantino = true;
97  }
98 
99  /* Get Geant4 pre- and post-step points */
100  G4StepPoint * prePoint = aStep->GetPreStepPoint();
101  G4StepPoint * postPoint = aStep->GetPostStepPoint();
102 
103  switch (prePoint->GetStepStatus())
104  {
105  //-----------------
106  case fGeomBoundary:
107  //-----------------
108  case fUndefined:
109  if (! hit) hit = new PHG4Hitv1();
110 
111  /* Set hit location (space point) */
112  hit->set_x( 0, prePoint->GetPosition().x() / cm);
113  hit->set_y( 0, prePoint->GetPosition().y() / cm );
114  hit->set_z( 0, prePoint->GetPosition().z() / cm );
115 
116  /* Set hit time */
117  hit->set_t( 0, prePoint->GetGlobalTime() / nanosecond );
118 
119  //set the track ID
120  hit->set_trkid(aTrack->GetTrackID());
121  /* set intial energy deposit */
122  hit->set_edep( 0 );
123  hit->set_eion( 0 );
124 
125  // here we set what is common for scintillator and absorber hits
126  if ( G4VUserTrackInformation* p = aTrack->GetUserInformation() ) {
127  if ( PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p) ) {
128  hit->set_trkid(pp->GetUserTrackId());
129  hit->set_shower_id(pp->GetShower()->get_id());
130  }
131  }
132  break;
133  //-----------------
134  default:
135  break;
136  }
137 
138  /* Update exit values- will be overwritten with every step until
139  * we leave the volume or the particle ceases to exist */
140  hit->set_x( 1, postPoint->GetPosition().x() / cm );
141  hit->set_y( 1, postPoint->GetPosition().y() / cm );
142  hit->set_z( 1, postPoint->GetPosition().z() / cm );
143 
144  hit->set_t( 1, postPoint->GetGlobalTime() / nanosecond );
145 
146  /* sum up the energy to get total deposited */
147  hit->set_edep(hit->get_edep() + edep);
148  hit->set_eion(hit->get_eion() + eion);
149 
150  if (geantino) {
151  hit->set_edep(-1); // only energy=0 g4hits get dropped, this way geantinos survive the g4hit compression
152  hit->set_eion(-1);
153  }
154  if (edep > 0 /*&& (whichactive > 0 || absorbertruth > 0)*/) {
155  if ( G4VUserTrackInformation* p = aTrack->GetUserInformation() ) {
156  if ( PHG4TrackUserInfoV1* pp = dynamic_cast<PHG4TrackUserInfoV1*>(p) ) {
157  pp->SetKeep(1); // we want to keep the track
158  }
159  }
160  }
161  return true;
162 
163  }
164  else return false;
165 
166 }

188
