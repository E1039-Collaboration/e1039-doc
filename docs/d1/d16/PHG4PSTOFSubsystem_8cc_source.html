43 int PHG4PSTOFSubsystem::InitRunSubsystem(PHCompositeNode *topNode)
44 {
45  PHNodeIterator iter(topNode);
46  PHCompositeNode *dstNode = dynamic_cast<PHCompositeNode *>(iter.findFirst("PHCompositeNode", "DST"));
47 
48  // create detector
49  detector_ = new PHG4PSTOFDetector(topNode, GetParamsContainer(), Name());
50  detector_->SuperDetector(SuperDetector());
51  detector_->OverlapCheck(CheckOverlap());
52 
53  set<string> nodes;
54  if (GetParamsContainer()->GetParameters(-1)->get_int_param("active"))
55  {
56  PHNodeIterator dstIter(dstNode);
57  PHCompositeNode *DetNode = dynamic_cast<PHCompositeNode *>(dstIter.findFirst("PHCompositeNode", SuperDetector()));
58  if (!DetNode)
59  {
60  DetNode = new PHCompositeNode(SuperDetector());
61  dstNode->addNode(DetNode);
62  }
63  ostringstream nodename;
64  if (SuperDetector() != "NONE")
65  {
66  nodename << "G4HIT_" << SuperDetector();
67  }
68  else
69  {
70  nodename << "G4HIT_" << Name();
71  }
72  nodes.insert(nodename.str());
73  BOOST_FOREACH (string node, nodes)
74  {
75  PHG4HitContainer *g4_hits = findNode::getClass<PHG4HitContainer>(topNode, node.c_str());
76  if (!g4_hits)
77  {
78  g4_hits = new PHG4HitContainer(node);
79  DetNode->addNode(new PHIODataNode<PHObject>(g4_hits, node.c_str(), "PHObject"));
80  }
81  }
82  // create stepping action
83  steppingAction_ = new PHG4PSTOFSteppingAction(detector_, GetParamsContainer());
84  steppingAction_->Init();
85  }
86 
87  return 0;
88 }
