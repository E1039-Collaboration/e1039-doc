125 PHG4MapsDetector::ConstructMaps(G4LogicalVolume* trackerenvelope)
126 {
127  if(verbosity>0)
128  {
129  cout << " PHG4MapsDetector::ConstructMaps:" <<endl;
130  cout << " Constructing Layer " << layer << endl;
131  cout << endl;
132  }
133 
134 
135  //===================================
136  // Import the stave physical volume here
137  //===================================
138 
139  // import the staves from the gemetry file
140  std::unique_ptr<G4GDMLReadStructure> reader (new G4GDMLReadStructure());
141  G4GDMLParser gdmlParser(reader.get());
142  gdmlParser.Read(stave_geometry_file, false);
143 
144  // figure out which assembly we want
145  char assemblyname[500];
146  sprintf(assemblyname, "ITSUStave%i",layer);
147 
148  if (Verbosity())
149  cout << "Geting the stave assembly named " << assemblyname << endl;
150  G4AssemblyVolume* av_ITSUStave = reader->GetAssembly(assemblyname);
151 
152 // if (reader) delete reader;
153 
154  //=========================================
155  // Now we populate the whole layer with the staves
156  //==========================================
157 
158  // Step through all phi values
159 
160  // The phistep is determined by the arc length spacing needed bewtween staves, which depends on the stave type
161  // The numbers below are inferred from the arc length spacing in ITS.gdml, and reduced a small amount (< 1%) to make sure the default radius cases work
162  // These arcstep numbers matter, that is why they are hardcoded here - leave them alone!
163  double arcstep;
164  if(stave_type == 0)
165  {
166  arcstep = 12.25; // inner barrel, mm
167  }
168  else if(stave_type == 1)
169  {
170  arcstep = 56.8; // middle barrel, mm
171  }
172  else
173  {
174  // stave_type is 2
175  arcstep = 54.2; // outer barrel, mm
176  }
177 
178  if(N_staves < 0)
179  {
180  // The user did not specify how many staves to use for this layer, so we calculate the best value
181  // We choose a phistep that fits N_staves into this radius, but with an arclength separation of AT LEAST arcstep
182  // ideally, the radius would be chosen so that numstaves = N_staves exactly, to get the closest spacing of staves possible without overlaps
183  double numstaves = 2.0 * M_PI * layer_nominal_radius / arcstep; // this is just to print out
184  N_staves = int(2.0 * M_PI * layer_nominal_radius / arcstep); // this is the number of staves used
185 
186  // Also suggest the ideal radius for this layer
187  if (verbosity > 0)
188  {
189  cout << " Calculated N_staves for layer " << layer
190  << " layer_nominal_radius " << layer_nominal_radius
191  << " ITS arcstep " << arcstep
192  << " circumference divided by arcstep " << numstaves
193  << " N_staves " << N_staves
194  << endl;
195  cout << "A radius for this layer of " << (double) N_staves * arcstep / (2.0 * M_PI) + 0.01 << " or "
196  << (double) (N_staves+1) * arcstep / (2.0 * M_PI) + 0.01 << " would produce perfect stave spacing" << endl;
197  }
198  }
199  phistep = 2.0 * M_PI / (double) N_staves; // this produces even stave spacing
200  double z_location = 0.0;
201 
202  // The tilt for stave type 0 (usually layers 0-2) is phitilt = 0.304; // radians, equivalent to 17.4 degrees, now input from parameters
203  if(stave_type != 0)
204  phitilt = 0.0;
205 
206  if (Verbosity())
207  {
208  cout << " layer " << layer
209  << " layer_nominal_radius " << layer_nominal_radius
210  << " N_staves " << N_staves
211  << " phistep " << phistep
212  << " phtilt " << phitilt
213  << endl;
214  }
215 
216  // The stave starts out at (0,0,0) and oriented so that the sensors face upward in y
217  // So we need to rotate the sensor 90 degrees before placing it using phi_offset
218  // note that the gdml file uses a negative phi_offset - different coord system, apparently - the following works
219  double phi_offset = M_PI /2.0;
220 
221  for (int iphi=0; iphi<N_staves; iphi++)
222  {
223  // Place the ladder segment envelopes at the correct z and phi
224  // This is the azimuthal angle at which we place the stave
225  G4double phi_rotation = (double) iphi * phistep;
226 
227  G4RotationMatrix Ra;
228  G4ThreeVector Ta;
229 
230  if(verbosity >0)
231  cout << "phi_offset = " << phi_offset << " iphi " << iphi << " phi_rotation = " << phi_rotation << " phitilt " << phitilt << endl;
232 
233  // It is first rotated in phi by the azimuthal angle phi_rotation, plus the 90 degrees needed to point the face of the sensor at the origin, plus the tilt (if a tilt is appropriate)
234 
235  Ra.rotateZ(phi_rotation + phi_offset + phitilt); // note - if this is layer 0-2, phitilt is the additional tilt for clearance. Otherwise it is zero
236  // Then translated as follows
237 
238  Ta.setX(layer_nominal_radius * cos(phi_rotation));
239  Ta.setY(layer_nominal_radius * sin(phi_rotation)) ;
240  Ta.setZ( z_location );
241 
242  if(verbosity > 0)
243  cout << " iphi " << iphi << " phi_rotation " << phi_rotation
244  << " x " << layer_nominal_radius * cos(phi_rotation)
245  << " y " << layer_nominal_radius * sin(phi_rotation)
246  << " z " << z_location
247  << endl;
248 
249 
250  G4Transform3D Tr(Ra,Ta);
251 
252  av_ITSUStave->MakeImprint(trackerenvelope, Tr, 0, overlapcheck);
253  }
254 
255  if(verbosity > 0)
256  cout << "This layer has a total of " << N_staves << " staves" << endl;
257 
258  SetDisplayProperty(av_ITSUStave);
259 
260  return 0;
