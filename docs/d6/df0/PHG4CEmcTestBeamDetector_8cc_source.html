174

176  {

192  "Plate_log_W",
193  0, 0, 0));
194  block_logic.push_back(new G4LogicalVolume(block_sc,
195  ScintiMaterial,
196  "Plate_log_Sc",
197  0, 0, 0));
198  G4VisAttributes* matVis = new G4VisAttributes();
199  G4VisAttributes* matVis1 = new G4VisAttributes();
200  matVis->SetVisibility(true);
201  matVis->SetForceSolid(true);
202  matVis->SetColour(G4Colour::Red());
203  matVis1->SetVisibility(true);
204  matVis1->SetForceSolid(true);
205  matVis1->SetColour(G4Colour::Green());
206  block_logic[0]->SetVisAttributes(matVis);
207  block_logic[1]->SetVisAttributes(matVis1);
208 
209  // here is the math to set the positions of those tiles inside the sandwich. The center of the coordinate system is in the center
210  // of the sandwich (w -> thickness of tungsten layer, sc_a -> thickness of actvie scinti, sc_p -> thickness of passive scinti)
211  // distance to the bottom of the sandwich:
212  // -(w + sc_a + sc_p + w)/2 = -w/2 -sc_a/2 -sc_p/2 -w/2
213 
214  // implement the absorber at the bottom and at the top of the sandwich
215  // moving up by w/2 for the lowest layer tungsten:
216  // -w/2 -sc_a/2 -sc_p/2 -w/2 + w/2 = -w/2 -sc_a/2 -sc_p/2 = -(w+sc)/2
217  // where sc_a+sc_p = sc = scintillator thickness (sc_dimension[1])
218 
219  sandwich_vol[0] = new G4PVPlacement(0, G4ThreeVector(0, -(w_dimension[1]+sc_dimension[1])/ 2., 0),
220  block_logic[0],
221  "CEmc_W_plate_down",
222  sandwich, 0, 0, overlapcheck);
223 
224  // top of the sandwich - starting at the bottom and add the tungsten and scintillator layer and half tungsten
225  // -w/2 -sc/2 -w/2 + w + sc + w/2 = +sc/2 +w/2 = (w+sc)/2
226  sandwich_vol[1] = new G4PVPlacement(0, G4ThreeVector(0, (w_dimension[1]+sc_dimension[1])/ 2., 0),
227  block_logic[0],
228  "CEmc_W_plate_up",
229  sandwich, 0, 0, overlapcheck);
230 
231  // since we split the scintillator into an active and passive part to accomodate
232  // for the scintillator fibers not occupying 100% of the volume.
233  // this is mocked up by 2 separate layers of plastic, active scintillator is the middle layer
234  // again going to the bottom of the sandwich box (sc_a -> active sc, sc_p -> passive sc)
235  // -w/2 -sc_a/2 -sc_p/2 -w/2
236  // -(w + sc_a + sc_p + w)/2, adding the w layer and half of the sc_a to get to the middle of the sc_a layer:
237  // -w/2 -sc_a/2 - sc_p/2 -w/2 + w + sc_a/2 = -sc_p/2
238  // if fraction of active sc is 1, sc_passive_thickness is zero
239  sandwich_vol[2] = new G4PVPlacement(0, G4ThreeVector(0, -sc_passive_thickness/ 2., 0),
240  block_logic[1],
241  "CEmc_Scinti_plate",
242  sandwich, 0, 0, overlapcheck);
243 
244 
245  if (sc_passive_thickness > 0)
246  {
247  G4VisAttributes* matVis2 = new G4VisAttributes();
248  matVis1->SetVisibility(true);
249  matVis1->SetForceSolid(true);
250  matVis1->SetColour(G4Colour::Blue());
251  block_passive_sc = new G4Box("Passive_Scinti_box",
252  sc_dimension[0] / 2., sc_passive_thickness / 2., sc_dimension[2] / 2.);
253  block_logic.push_back( new G4LogicalVolume(block_passive_sc,
254  ScintiMaterial,
255  "Plate_log_Passive_Sc",
256  0, 0, 0));
257  block_logic[2]->SetVisAttributes(matVis2);
258  // here we go again - bottome of sandwich box is
259  // -(w + sc_a + sc_p +w)/2, now add w and sc_a and half of sc_p:
260  // -w/2 -sc_a/2 - sc_p/2 -w/2 + w + sc_a + sc_p/2 = sc_a/2
261  sandwich_vol[3] = new G4PVPlacement(0, G4ThreeVector(0, sc_active_thickness / 2., 0),
262  block_logic[2],
263  "CEmc_Passive_Si_plate",
264  sandwich, 0, 0, overlapcheck);
265  }
266  else
267  {
268  sandwich_vol[3] = NULL;
269  }
270  return 0;

273 void
274 PHG4CEmcTestBeamDetector::CalculateGeometry()
275 {
276  // calculate the inner/outer radius of the g4tub using the test setup numbers
277  // 1mm scintillator, 0.5mm tungsten, 0.25 mm gap at the back of the detector
278  // the geometry with figure is explained in our wiki:
279  // https://www.phenix.bnl.gov/WWW/offline/wikioff/index.php/CEmc
280  // get the alpha angle via law of cos (a is the 0.25 mm gap):
281  // a^2 = b^2+c^2 - 2bc*cos(alpha)
282  // cos(alpha) = (b^2+c^2 - a^2)/2bc
283  // with b=c
284  // cos(alpha) = 1-(a^2/2b^2)
285  alpha = acos(1 - (gap * gap) / (2 * plate_x * plate_x));
286  // inner radius
287  inner_radius = sandwich_thickness / tan(alpha);
288  outer_radius = sqrt((inner_radius + plate_x) * (inner_radius + plate_x) + sandwich_thickness * sandwich_thickness);
289 
290  // twice no_overlap to apply safety margin also for towers
291  // angular coverage of 1 tower is 12 sandwiches = sandwiches_per_tower*alpha
292  tower_angular_coverage = sandwiches_per_tower * alpha + 0.00005 * deg; // safety margin added to remove 142 nm overlap
293  // cemc prototype has 7 towers
294  cemc_angular_coverage = num_towers * tower_angular_coverage;
295  return;
296 }
