364 double
365 PHG4Prototype2InnerHcalSteppingAction::GetLightCorrection(const double r) const
366 {
367  double m = (light_balance_outer_corr - light_balance_inner_corr) / (light_balance_outer_radius - light_balance_inner_radius);
368  double b = light_balance_inner_corr - m * light_balance_inner_radius;
369  double value = m * r + b;
370  if (value > 1.0) return 1.0;
371  if (value < 0.0) return 0.0;
372 
373  return value;
374 }
