72 void Plane::update()
73 {
74  xc = x0 + deltaX;
75  yc = y0 + deltaY;
76  zc = z0 + deltaZ;
77 
78  rX = thetaX + rotX;
79  rY = thetaY + rotY;
80  rZ = thetaZ + rotZ;
81 
82  sintheta = sin(angleFromVert + rZ);
83  costheta = cos(angleFromVert + rZ);
84  tantheta = tan(angleFromVert + rZ);
85 
86  wc = getW(xc, yc);
87 
88  rotM[0][0] = cos(rZ)*cos(rY);
89  rotM[0][1] = cos(rZ)*sin(rX)*sin(rY) - cos(rX)*sin(rZ);
90  rotM[0][2] = cos(rX)*cos(rZ)*sin(rY) + sin(rX)*sin(rZ);
91  rotM[1][0] = sin(rZ)*cos(rY);
92  rotM[1][1] = sin(rZ)*sin(rX)*sin(rY) + cos(rZ)*cos(rX);
93  rotM[1][2] = sin(rZ)*sin(rY)*cos(rX) - cos(rZ)*sin(rX);
94  rotM[2][0] = -sin(rY);
95  rotM[2][1] = cos(rY)*sin(rX);
96  rotM[2][2] = cos(rY)*cos(rX);
97 
98  uVec[0] = cos(angleFromVert);
99  uVec[1] = sin(angleFromVert);
100  uVec[2] = 0.;
101 
102  vVec[0] = -sin(angleFromVert);
103  vVec[1] = cos(angleFromVert);
104  vVec[2] = 0.;
105 
106  nVec[0] = 0.;
107  nVec[1] = 0.;
108  nVec[2] = 1.;
109 
110  double temp[3];
111  for(int i = 0; i < 3; ++i) temp[i] = uVec[i];
112  for(int i = 0; i < 3; ++i)
113  {
114  uVec[i] = 0.;
115  for(int j = 0; j < 3; ++j)
116  {
117  uVec[i] += rotM[i][j]*temp[j];
118  }
119  }
120 
121  for(int i = 0; i < 3; ++i) temp[i] = vVec[i];
122  for(int i = 0; i < 3; ++i)
123  {
124  vVec[i] = 0.;
125  for(int j = 0; j < 3; ++j)
126  {
127  vVec[i] += rotM[i][j]*temp[j];
128  }
129  }
130 
131  nVec[0] = uVec[1]*vVec[2] - vVec[1]*uVec[2];
132  nVec[1] = uVec[2]*vVec[0] - vVec[2]*uVec[0];
133  nVec[2] = uVec[0]*vVec[1] - vVec[0]*uVec[1];
134 }

136 double Plane::intercept(double tx, double ty, double x0_track, double y0_track) const
137 {
138  //double mom[3] = {tx, ty, 1.};
139  //double pos[3] = {x0_track, y0_track, 0.};
140 
141  double det = -(tx*nVec[0] + ty*nVec[1] + nVec[2]);
142  double dpos[3] = {x0_track - xc, y0_track - yc, -zc};
143 
144  double vcp[3];
145  vcp[0] = vVec[1] - vVec[2]*ty;
146  vcp[1] = vVec[2]*tx - vVec[0];
147  vcp[2] = vVec[0]*ty - vVec[1]*tx;
148 
149  //LogInfo(detectorID << " " << detectorName << " " << tx << " " << ty << " " << x0_track << " " << y0_track << " " << -(vcp[0]*dpos[0] + vcp[1]*dpos[1] + vcp[2]*dpos[2])/det);
150  return -(vcp[0]*dpos[0] + vcp[1]*dpos[1] + vcp[2]*dpos[2])/det + wc;
