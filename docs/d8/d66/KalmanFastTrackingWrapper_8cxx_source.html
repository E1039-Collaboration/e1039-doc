142 int KalmanFastTrackingWrapper::process_event(PHCompositeNode* topNode) {
143 
144  if(Verbosity() >= Fun4AllBase::VERBOSITY_SOME)
145  std::cout << "Entering KalmanFastTrackingWrapper::process_event: " << _event << std::endl;
146 
147  ResetEvalVars();
148 
149  if(!_event_header) {
150  LogDebug("!_event_header");
151  return Fun4AllReturnCodes::ABORTRUN;
152  }
153 
154  if(!_spill_map) {
155  LogDebug("!_spill_map");
156  return Fun4AllReturnCodes::ABORTRUN;
157  }
158 
159  if(!_hit_vector) {
160  LogDebug("!_hit_vector");
161  return Fun4AllReturnCodes::ABORTRUN;
162  }
163 
164  SRawEvent* sraw_event = BuildSRawEvent();
165  _rawEvent = sraw_event;
166 
167  _recEvent = new SRecEvent();
168 
169  _recEvent->setRecStatus(fastfinder->setRawEvent(sraw_event));
170 
171  _recEvent->setRawEvent(_rawEvent);
172 
173  std::list<Tracklet>& rec_tracklets = fastfinder->getFinalTracklets();
174 
175  for(std::list<Tracklet>::iterator iter = rec_tracklets.begin(); iter != rec_tracklets.end(); ++iter)
176  {
177  iter->calcChisq();
178  iter->print();
179 
180  SRecTrack recTrack = iter->getSRecTrack();
181  _recEvent->insertTrack(recTrack);
182  }
183 
184 
185  _tout->Fill();
186 
187  if(Verbosity() >= Fun4AllBase::VERBOSITY_SOME)
188  std::cout << "Leaving KalmanFastTrackingWrapper::process_event: " << _event << std::endl;
189  ++_event;
190 
191  return Fun4AllReturnCodes::EVENT_OK;
192 }
