643 bool PHG4MapsCellReco::lines_intersect(
644  double ax,
645  double ay,
646  double bx,
647  double by,
648  double cx,
649  double cy,
650  double dx,
651  double dy,
652  double* rx, // intersection point (output)
653  double* ry
654  )
655 {
656 
657  // Find if a line segment limited by points A and B
658  // intersects line segment limited by points C and D.
659  // First check if an infinite line defined by A and B intersects
660  // segment (C,D). If h is from 0 to 1 line and line segment intersect
661  // Then check in intersection point is between C and D
662 
663  double ex = bx - ax; // E=B-A
664  double ey = by - ay;
665  double fx = dx - cx; // F=D-C
666  double fy = dy - cy;
667  double px = -ey; // P
668  double py = ex;
669 
670  double bottom = fx * px + fy * py; // F*P
671  double gx = ax - cx; // A-C
672  double gy = ay - cy;
673  double top = gx * px + gy * py; // G*P
674 
675  double h = 99999.;
676  if (bottom != 0.)
677  {
678  h = top / bottom;
679  }
680 
681 //intersection point R = C + F*h
682  if (h > 0. && h < 1.)
683  {
684  *rx = cx + fx * h;
685  *ry = cy + fy * h;
686  //cout << " line/segment intersection coordinates: " << *rx << " " << *ry << endl;
687  if ((*rx > ax && *rx > bx) || (*rx < ax && *rx < bx) || (*ry < ay && *ry < by) || (*ry > ay && *ry > by))
688  {
689  //cout << " NO segment/segment intersection!" << endl;
690  return false;
691  }
692  else
693  {
694  //cout << " segment/segment intersection!" << endl;
695  return true;
696  }
697  }
698 
699  return false;
700 }

702 bool PHG4MapsCellReco::line_and_rectangle_intersect(
703  double ax,
704  double ay,
705  double bx,
706  double by,
707  double cx,
708  double cy,
709  double dx,
710  double dy,
711  double* rr // length of the line segment inside the rectangle (output)
712  )
713 {
714 
715  // find if a line isegment limited by points (A,B)
716  // intersects with a rectangle defined by two
717  // corner points (C,D) two other points are E and F
718  // E--------D
719  // | |
720  // | |
721  // C--------F
722 
723  if (cx > dx || cy > dy)
724  {
725  cerr << "ERROR: Bad rectangle definition!" << endl;
726  return false;
727  }
728 
729  double ex = cx;
730  double ey = dy;
731  double fx = dx;
732  double fy = cy;
733  double rx = 99999.;
734  double ry = 99999.;
735 
736  vector<double> vx;
737  vector<double> vy;
738 
739  bool i1 = lines_intersect(ax, ay, bx, by, cx, cy, fx, fy, &rx, &ry);
740  if (i1)
741  {
742  vx.push_back(rx);
743  vy.push_back(ry);
744  }
745  bool i2 = lines_intersect(ax, ay, bx, by, fx, fy, dx, dy, &rx, &ry);
746  if (i2)
747  {
748  vx.push_back(rx);
749  vy.push_back(ry);

752  if (i3)
753  {
754  vx.push_back(rx);
755  vy.push_back(ry);
756  }
757  bool i4 = lines_intersect(ax, ay, bx, by, cx, cy, ex, ey, &rx, &ry);
758  if (i4)
759  {
760  vx.push_back(rx);
761  vy.push_back(ry);
762  }
763 
764 //cout << "Rectangle intersections: " << i1 << " " << i2 << " " << i3 << " " << i4 << endl;
765 //cout << "Number of intersections = " << vx.size() << endl;
766 
767  *rr = 0.;
768  if (vx.size() == 2)
769  {
770  *rr = sqrt( (vx[0] - vx[1]) * (vx[0] - vx[1]) + (vy[0] - vy[1]) * (vy[0] - vy[1]) );
771 // cout << "Length of intersection = " << *rr << endl;
772  }
773  if (vx.size() == 1)
774  {
775  // find which point (A or B) is within the rectangle
776  if (ax > cx && ay > cy && ax < dx && ay < dy) // point A is inside the rectangle
777  {
778  //cout << "Point A is inside the rectangle." << endl;

780  }
781  if (bx > cx && by > cy && bx < dx && by < dy) // point B is inside the rectangle
782  {
783  //cout << "Point B is inside the rectangle." << endl;
784  *rr = sqrt((vx[0] - bx) * (vx[0] - bx) + (vy[0] - by) * (vy[0] - by));
785  }
786  }
787 
788  if (i1 || i2 || i3 || i4)
789  {
790  return true;
791  }
792  return false;

820 double PHG4MapsCellReco::sA(double r, double x, double y)
821 {
822  // Uses analytic formula for the integral of a circle between limits set by the corner of a rectangle
823  // It is called repeatedly to find the overlap area between the circle and rectangle
824  // I found this code implementing the integral on a web forum called "ars technica",
825  // https://arstechnica.com/civis/viewtopic.php?t=306492
826  // posted by "memp"
827 
828  double a;
829 
830  if (x < 0)
831  {
832  return -sA(r, -x, y);
833  }
834 
835  if (y < 0)
836  {
837  return -sA(r, x, -y);
838  }
839 
840  if (x > r)
841  {
842  x = r;
843  }
844 
845  if (y > r)
846  {
847  y = r;
848  }
849 
850  if (x*x + y*y > r*r)
851  {
852  a = r*r*asin(x/r) + x*sqrt(r*r-x*x)
853  + r*r*asin(y/r) + y*sqrt(r*r-y*y)
854  - r*r*M_PI_2;
855 
856  a *= 0.5;
857  }
858  else
859  {
860  a = x*y;
861  }
862 
863  return a;
864 }
