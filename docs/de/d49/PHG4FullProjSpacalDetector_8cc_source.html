499  + weighted_pDx2 * ( weighted_iy)
500  + weighted_pDy1 * tan(g_tower.pAlp1);
501  const double weighted_pDx34 = weighted_pDx3 * (1-weighted_iy)
502  + weighted_pDx4 * (weighted_iy)
503  + weighted_pDy1 * tan(g_tower.pAlp2);
504 
505  G4Vector3D v1 = G4Vector3D(weighted_pDx12, weighted_pDy1, 0)
506  - v_zshift;
507  G4Vector3D v2 = G4Vector3D(weighted_pDx34, weighted_pDy2, 0)
508  + v_zshift;
509 
510  G4Vector3D vector_fiber = (v2 - v1);
511  vector_fiber *= (vector_fiber.mag() - get_geom_v3()->get_fiber_outer_r())
512  / vector_fiber.mag(); // shrink by fiber boundary protection
513  G4Vector3D center_fiber = (v2 + v1) / 2;
514 
515  // convert to Geant4 units
516  vector_fiber *= cm;
517  center_fiber *= cm;
518 
519  const G4double fiber_length = vector_fiber.mag();
520 
521  stringstream ss;
522  ss << string("_Tower") << g_tower.id;
523  ss << "_x" << ix;
524  ss << "_y" << iy;
525  G4LogicalVolume *fiber_logic = Construct_Fiber(fiber_length,
526  ss.str());
527 
528  if (get_geom_v3()->get_construction_verbose() >= 3)
529  cout << "PHG4FullProjSpacalDetector::Construct_Fibers::" << GetName()
530  << " - constructed fiber " << fiber_ID << ss.str() //
531  << ", Length = " << fiber_length << "mm, " //
532  << "x = " << center_fiber.x() << "mm, " //
533  << "y = " << center_fiber.y() << "mm, " //
534  << "z = " << center_fiber.z() << "mm, " //
535  << "vx = " << vector_fiber.x() << "mm, " //
536  << "vy = " << vector_fiber.y() << "mm, " //
537  << "vz = " << vector_fiber.z() << "mm, " //
538  << endl;
539 
540  const G4double rotation_angle = G4Vector3D(0, 0, 1).angle(
541  vector_fiber);
542  const G4Vector3D rotation_axis =
543  rotation_angle == 0 ?
544  G4Vector3D(1, 0, 0) : G4Vector3D(0, 0, 1).cross(vector_fiber);
545 
546  G4Transform3D fiber_place(
547  G4Translate3D(center_fiber.x(), center_fiber.y(),
548  center_fiber.z())
549  * G4Rotate3D(rotation_angle, rotation_axis));
550 
551  stringstream name;
552  name << GetName() + string("_Tower") << g_tower.id << "_fiber"
553  << ss.str();
554 
555  const bool overlapcheck_fiber = overlapcheck
556  and (get_geom_v3()->get_construction_verbose() >= 3);
557  G4PVPlacement * fiber_physi = new G4PVPlacement(fiber_place,
558  fiber_logic, G4String(name.str().c_str()), LV_tower, false,
559  fiber_ID, overlapcheck_fiber);
560  fiber_vol[fiber_physi] = fiber_ID;
561 
562  assert(gdml_config);
563  gdml_config->exclude_physical_vol(fiber_physi);
564 
565  ++fiber_cnt;
566  }
567  }
568 
569  if (get_geom_v3()->get_construction_verbose() >= 3)
570  cout << "PHG4FullProjSpacalDetector::Construct_Fibers::" << GetName()
571  << " - constructed tower ID " << g_tower.id << " with " << fiber_cnt
572  << " fibers" << endl;
573 
574  return fiber_cnt;
575 }

583 
584  std::stringstream sout;

586  const G4String sTowerID(sout.str());
587 
588  //Processed PostionSeeds 1 from 1 1
589 
590  G4Trap* block_solid = new G4Trap(
591  /*const G4String& pName*/G4String(GetName().c_str()) + sTowerID,
592  g_tower.pDz * cm, // G4double pDz,
593  g_tower.pTheta * rad, g_tower.pPhi * rad, // G4double pTheta, G4double pPhi,
594  g_tower.pDy1 * cm, g_tower.pDx1 * cm, g_tower.pDx2 * cm, // G4double pDy1, G4double pDx1, G4double pDx2,
595  g_tower.pAlp1 * rad, // G4double pAlp1,
596  g_tower.pDy2 * cm, g_tower.pDx3 * cm, g_tower.pDx4 * cm, // G4double pDy2, G4double pDx3, G4double pDx4,
597  g_tower.pAlp2 * rad // G4double pAlp2 //
598  );
599 
600  G4Material * cylinder_mat = G4Material::GetMaterial(
601  get_geom_v3()->get_absorber_mat());
602  assert(cylinder_mat);
603 
604  G4LogicalVolume * block_logic = new G4LogicalVolume(block_solid, cylinder_mat,
605  G4String(G4String(GetName()) + string("_Tower") + sTowerID), 0, 0,
606  nullptr);
607 
608  G4VisAttributes* VisAtt = new G4VisAttributes();
609 // PHG4Utils::SetColour(VisAtt, "W_Epoxy");
610  VisAtt->SetColor(.3, .3, .3, .3);
611  VisAtt->SetVisibility(
612  get_geom_v3()->is_azimuthal_seg_visible() or get_geom_v3()->is_virualize_fiber());
613  VisAtt->SetForceSolid(not get_geom_v3()->is_virualize_fiber());
614  block_logic->SetVisAttributes(VisAtt);
615

617 
618  int fiber_count = 0;
619 
620  if (get_geom_v3()->get_config() == SpacalGeom_t::kFullProjective_2DTaper)
621  {
622  fiber_count = Construct_Fibers(g_tower, block_logic);
623

627  << fiber_count << " fibers using Construct_Fibers" << endl;
628  }
629  else if (get_geom_v3()->get_config()
630  == SpacalGeom_t::kFullProjective_2DTaper_SameLengthFiberPerTower)
631  {
632  fiber_count = Construct_Fibers_SameLengthFiberPerTower(g_tower,
633  block_logic);
634 
635  if (get_geom_v3()->get_construction_verbose() >= 2)
636  cout << "PHG4FullProjSpacalDetector::Construct_Tower::" << GetName()
637  << " - constructed tower ID " << g_tower.id << " with "
638  << fiber_count
639  << " fibers using Construct_Fibers_SameLengthFiberPerTower" << endl;
640  }
641  else
642  {
643 
644  G4ExceptionDescription message;
645  message << "can not recognize configuration type " << get_geom_v3()->get_config();
646 
647  G4Exception("PHG4FullProjSpacalDetector::Construct_Tower", "Wrong",
648  FatalException, message, "");
649  }
