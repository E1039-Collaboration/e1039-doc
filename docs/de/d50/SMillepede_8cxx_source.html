132  {

134

287 void SMillepede::initMillepede(std::string configFileName)
288 {
289  //Define the dimension of arrays in millepede
290  int nGlobal = MILLEPEDE::NGLB;
291  int nLocal = MILLEPEDE::NPARTRK;
292  int nStdev = MILLEPEDE::NSTDEV;
293  int iPrint = 1;
294 
295  initgl_(&nGlobal, &nLocal, &nStdev, &iPrint);
296 
297  //Set parameter initial value and resolution parameter
298  double err_z[MILLEPEDE::NPLAN], err_phi[MILLEPEDE::NPLAN], err_w[MILLEPEDE::NPLAN];
299  for(int i = 1; i <= MILLEPEDE::NPLAN; ++i)
300  {
301  err_z[i-1] = 0.1;
302  err_phi[i-1] = 0.005;
303  err_w[i-1] = 0.05;
304  }
305 
306  std::fstream conf;
307  conf.open(configFileName.c_str(), std::ios::in);
308  if(conf)
309  {
310  int detectorID;
311  double err1, err2, err3;
312  char buf[200];
313  while(conf.getline(buf, 200))
314  {
315  std::istringstream stringBuf(buf);
316  stringBuf >> detectorID >> err1 >> err2 >> err3;
317 
318  err_z[detectorID-1] = err1;
319  err_phi[detectorID-1] = err2;
320  err_w[detectorID-1] = err3;
321  }
322  }
323  conf.close();
324 
325  for(int i = 1; i <= MILLEPEDE::NPLAN; i++)
326  {
327  setDetectorParError(i, 0, err_z[i-1]);
328  setDetectorParError(i, 1, err_phi[i-1]);
329  setDetectorParError(i, 2, err_w[i-1]);
330  }
331 
332  //fix one detector as-is, it needs to be done here as it sets the center value as well
333  fixDetectorParameter(p_geomSvc->getDetectorID("D2X"), 0, 0.);
334  fixDetectorParameter(p_geomSvc->getDetectorID("D2X"), 1, 0.);
335  fixDetectorParameter(p_geomSvc->getDetectorID("D2X"), 2, 0.);
336 
337  // Now pass the info above to millepede
338  parglo_(par_align);
339  for(int i = 0; i < MILLEPEDE::NGLB; i++)
340  {
341  int index = i+1;
342  parsig_(&index, &err_align[i]);
343  }
344 
345  //Constrains
346  //1. Fix adjacent planes to be same
347  for(int i = 1; i <= MILLEPEDE::NPLAN; i += 2)
348  {
349  constrainDetectors(i, i+1, 0);
350  constrainDetectors(i, i+1, 1);
351  //constrainDetectors(i, i+1, 2);
352 
353  //for station 1 or 3+/-, w offset is also fixed together
354  if((i > 6 && i < 13) || i > 18) constrainDetectors(i, i+1, 2);
355  }
356 
357  //2. Fix all 6 planes of station 1/3+/3- to be the same in z and phi
358  for(int i = 7; i < 10; i += 2)
359  {
360  constrainDetectors(i, i+1, 0);
361  constrainDetectors(i, i+1, 1);
362  }
363 
364  for(int i = 20; i < 23; i += 2)
365  {
366  constrainDetectors(i, i+1, 0);
367  constrainDetectors(i, i+1, 1);
368  }
369 
370  for(int i = 26; i < 29; i += 2)
371  {
372  constrainDetectors(i, i+1, 0);
373  constrainDetectors(i, i+1, 1);
374  }
375 
376  //3. the global offsets of station 1, 3+ and 3- is fixed
377  float rhs = 0.;
378  float dercs[MILLEPEDE::NGLB];
379 
380  //1
381  for(int k = 0; k < MILLEPEDE::NGLB; k++) dercs[k] = 0.;
382  dercs[MILLEPEDE::NPARPLAN*6 + 2] = 1.;
383  dercs[MILLEPEDE::NPARPLAN*8 + 2] = -(p_geomSvc->getCostheta(7) + p_geomSvc->getCostheta(11));
384  dercs[MILLEPEDE::NPARPLAN*10 + 2] = 1.;
385 
386  constf_(dercs, &rhs);
387 
388  //3+
389  for(int k = 0; k < MILLEPEDE::NGLB; k++) dercs[k] = 0.;
390  dercs[MILLEPEDE::NPARPLAN*18 + 2] = 1.;
391  dercs[MILLEPEDE::NPARPLAN*20 + 2] = -(p_geomSvc->getCostheta(19) + p_geomSvc->getCostheta(23));
392  dercs[MILLEPEDE::NPARPLAN*22 + 2] = 1.;
393 
394  constf_(dercs, &rhs);
395 
396  //3-
397  for(int k = 0; k < MILLEPEDE::NGLB; k++) dercs[k] = 0.;
398  dercs[MILLEPEDE::NPARPLAN*24 + 2] = 1.;
399  dercs[MILLEPEDE::NPARPLAN*26 + 2] = -(p_geomSvc->getCostheta(25) + p_geomSvc->getCostheta(29));
400  dercs[MILLEPEDE::NPARPLAN*28 + 2] = 1.;
401 
402  constf_(dercs, &rhs);
403 
404  //Fix global positions
405  /*
406  for(int k = 0; k < MILLEPEDE::NGLB; k++) dercs[k] = 0.;
407  for(int i = 0; i < MILLEPEDE::NPLAN; i++) dercs[i*MILLEPEDE::NPARPLAN + 0] = 1.;
408  constf_(dercs, &rhs);
409 
410  for(int k = 0; k < MILLEPEDE::NGLB; k++) dercs[k] = 0.;
411  for(int i = 0; i < MILLEPEDE::NPLAN; i++) dercs[i*MILLEPEDE::NPARPLAN + 1] = 1.;
412  constf_(dercs, &rhs);
413 
414  //for(int k = 0; k < MILLEPEDE::NGLB; k++) dercs[k] = 0.;
415  //for(int i = 0; i < MILLEPEDE::NPLAN; i++) dercs[i*MILLEPEDE::NPARPLAN + 2] = 1.;
416  //constf_(dercs, &rhs);
417  */
418

441 void SMillepede::setSingleTrack()
442 {
443  using namespace MILLEPEDE;
444 
445  //Global and local derivatives
446  float dergb[NGLB];
447  float derlc[NPARTRK];
448  float meas;
449  float sigma;
450 
451  //Fill the nodes to derivative arrays
452  for(std::vector<MPNode>::iterator node = nodes.begin(); node != nodes.end(); ++node)
453  {
454  if(!node->isValid()) continue;
455 
456  //Initialization of all parameters
457  zerloc_(dergb, derlc);
458 
459  //Get measurements
460  int index = node->detectorID - 1;
461  meas = node->meas;
462  sigma = node->sigma;
463 
464  //Fill local derivarives
465  derlc[0] = node->dwdx;
466  derlc[1] = node->dwdy;
467  derlc[2] = node->dwdtx;
468  derlc[3] = node->dwdty;
469 
470  //Fill global derivatives
471  if(index >= 0)
472  {
473  dergb[NPARPLAN*index + 0] = node->dwdz;
474  dergb[NPARPLAN*index + 1] = node->dwdphi;
475  dergb[NPARPLAN*index + 2] = node->dwdw;
476  }
477 
478  //Book the local/global derivatives, measurement and error
479  equloc_(dergb, derlc, &meas, &sigma);
480  }
481 
482  //Perform local track fit
483  fitloc_();
484 }
