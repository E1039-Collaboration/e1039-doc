426 double
427 PHG4InnerHcalSteppingAction::GetLightCorrection(const double r) const
428 {
429  double m = (light_balance_outer_corr - light_balance_inner_corr) / (light_balance_outer_radius - light_balance_inner_radius);
430  double b = light_balance_inner_corr - m * light_balance_inner_radius;
431  double value = m * r + b;
432  if (value > 1.0) return 1.0;
433  if (value < 0.0) return 0.0;
434 
435  return value;
436 }
