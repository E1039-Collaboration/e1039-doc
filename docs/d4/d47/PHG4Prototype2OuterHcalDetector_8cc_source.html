397 int
398 PHG4Prototype2OuterHcalDetector::ConstructOuterHcal(G4LogicalVolume* hcalenvelope)
399 {
400  G4LogicalVolume* steel_plate = ConstructSteelPlate(hcalenvelope); // bottom steel plate
401  G4LogicalVolume* scintibox = NULL;
402  if (params->get_int_param("hi_eta"))
403  {
404  scintibox = ConstructScintillatorBoxHiEta(hcalenvelope);
405  }
406  else
407  {
408  scintibox = ConstructScintillatorBox(hcalenvelope);
409  }
410  double phi = 0.;
411  double phislat = 0.;
412  ostringstream name;
413  // the coordinate of the center of the bottom of the bottom steel plate
414  // to get the radius of the circle which is the center of the scintillator box
415  double bottom_xmiddle_steel_tile = (steel_plate_corner_lower_right.x()-steel_plate_corner_lower_left.x())/2.+steel_plate_corner_lower_left.x();
416  double bottom_ymiddle_steel_tile = steel_plate_corner_lower_right.y();
417  double middlerad = sqrt(bottom_xmiddle_steel_tile*bottom_xmiddle_steel_tile + bottom_ymiddle_steel_tile * bottom_ymiddle_steel_tile);
418  double philow = atan((bottom_ymiddle_steel_tile-scinti_gap/2.)/bottom_xmiddle_steel_tile);
419  double scintiangle = GetScintiAngle();
420  for (int i = 0; i < n_steel_plates; i++)
421  // for (int i = 0; i < 2; i++)
422  {
423  name.str("");
424  name << "OuterHcalSteel_" << i;
425  G4RotationMatrix *Rot = new G4RotationMatrix();
426  Rot->rotateZ(phi*rad);
427  G4ThreeVector g4vec(0,0,0);
428  outerhcalassembly->AddPlacedVolume(steel_plate,g4vec,Rot);
429  if (i > 0)
430  {
431  double ypos = sin(phi+philow) * middlerad;
432  double xpos = cos(phi+philow) * middlerad;
433  // the center of the scintillator is not the center of the inner hcal
434  // but depends on the tilt angle. Therefore we need to shift
435  // the center from the mid point
436  ypos += sin((-tilt_angle)/rad - phi);
437  xpos -= cos((-tilt_angle)/rad - phi);
438  name.str("");
439  name << "OuterHcalScintiBox_" << i;
440  Rot = new G4RotationMatrix();
441  Rot->rotateZ(scintiangle+phislat);
442  G4ThreeVector g4vec(xpos, ypos, 0);
443 
444  outerhcalassembly->AddPlacedVolume(scintibox,g4vec,Rot);
445  phislat += deltaphi;
446  }
447  phi += deltaphi;
448  }
449  return 0;
450 }
