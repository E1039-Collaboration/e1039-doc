263 int TrackExtrapolator::propagate()
264 {
265  travelLength = 0.;
266 
268  if(!(calcProp || calcLength))
269  {
270  return g4eMgr->Propagate(g4eState, g4eTarget, g4eMode);
271  }
272 
273 
275  if(calcProp)
276  {
277  for(int i = 0; i < 5; i++)
278  {
279  for(int j = 0; j < 5; j++)
280  {
281  g4eProp[i][j] = 0.;
282  if(i == j) g4eProp[i][j] = 1.;
283  }
284  }
285  }
286 
288  g4eMgr->InitTrackPropagation();
289  G4ThreeVector pos_before, pos_after;
290 
291  bool isLastStep = false;
292  while(!isLastStep)
293  {
294  if(calcLength) pos_before = g4eState->GetPosition();
295 
296  int ierr = g4eMgr->PropagateOneStep(g4eState, g4eMode);
297  if(ierr != 0)
298  {
299  return ierr;
300  }
301 
302  //calculate the propagation matrix
303  if(calcProp)
304  {
305  G4ErrorMatrix g4eProp_oneStep = g4eState->GetTransfMat();
306  G4ErrorMatrix g4eProp_temp = g4eProp_oneStep*g4eProp;
307  g4eProp = g4eProp_temp;
308  }
309 
310  //calculate the travel length
311  if(calcLength)
312  {
313  pos_after = g4eState->GetPosition();
314  if((pos_before[2] < FMAG_LENGTH*cm && pos_before[2] > 0) || (pos_after[2] < FMAG_LENGTH*cm && pos_after[2] > 0))
315  {
316  double length = (pos_before - pos_after).mag();
317  if(pos_after[2] < 0)
318  {
319  length *= fabs(pos_before[2]/(pos_before[2] - pos_after[2]));
320  }
321  else if(pos_before[2] > FMAG_LENGTH*cm)
322  {
323  length *= fabs((pos_after[2] - FMAG_LENGTH*cm)/(pos_before[2] - pos_after[2]));
324  }
325 
326  travelLength += length;
327  }
328  }
329 
330  //Check if the target plane is reached
331  isLastStep = g4eMgr->GetPropagator()->CheckIfLastStep(g4eState->GetG4Track());
332  }
333 
334  g4eMgr->GetPropagator()->InvokePostUserTrackingAction(g4eState->GetG4Track());
335  return 0;

405 void TrackExtrapolator::TRSDSC(int charge, G4ThreeVector mom_input, G4ThreeVector pos_input)
406 {
408  TVector3 mom(mom_input[0], mom_input[1], mom_input[2]);
409  TVector3 pos(pos_input[0], pos_input[1], pos_input[2]);
410 
411  mom *= 0.001;
412  pos *= 0.1;
413 
415  TVector3 DJ(1., 0., 0.);
416  TVector3 DK(0., 1., 0.);
417  TVector3 DI = DJ.Cross(DK);
418 
420  double p_inv = 1./mom.Mag();
421  double vp = mom.Dot(DJ)/mom.Dot(DI);
422  double wp = mom.Dot(DK)/mom.Dot(DI);
423  double lambda = TMath::Pi()/2. - mom.Theta();
424  //double phi = mom.Phi();
425 
426  TVector3 TVW;
427  TVW.SetX(1./sqrt(1. + vp*vp + wp*wp));
428  TVW.SetY(vp*TVW.X());
429  TVW.SetZ(wp*TVW.X());
430 
431  TVector3 TN;
432  for(int i = 0; i < 3; i++)
433  {
434  TN[i] = TVW[0]*DI[i] + TVW[1]*DJ[i] + TVW[2]*DK[i];
435  }
436 
437  double cosl = cos(lambda);
438  double cosl1 = 1./cosl;
439 
440  TVector3 UN(-TN.Y()*cosl1, TN.X()*cosl1, 0.);
441  TVector3 VN(-TN.Z()*UN.Y(), TN.Z()*UN.X(), cosl);
442 
443  double UJ = UN.Dot(DJ);
444  double UK = UN.Dot(DK);
445  double VJ = VN.Dot(DJ);
446  double VK = VN.Dot(DK);
447 
448  jac_sd2sc.Zero();
449  jac_sd2sc[0][0] = 1.;
450  jac_sd2sc[1][1] = TVW[0]*VJ;
451  jac_sd2sc[1][2] = TVW[0]*VK;
452  jac_sd2sc[2][1] = TVW[0]*UJ*cosl1;
453  jac_sd2sc[2][2] = TVW[0]*UK*cosl1;
454  jac_sd2sc[3][3] = UJ;
455  jac_sd2sc[3][4] = UK;
456  jac_sd2sc[4][3] = VJ;
457  jac_sd2sc[4][4] = VK;
458 
459  const G4Field* field = G4TransportationManager::GetTransportationManager()->GetFieldManager()->GetDetectorField();
460  if(charge != 0 && field)
461  {
462  double pos_cm[3];
463  pos_cm[0] = pos.X()*cm;
464  pos_cm[1] = pos.Y()*cm;
465  pos_cm[2] = pos.Z()*cm;
466 
467  double H_temp[3];
468  field->GetFieldValue(pos_cm, H_temp);
469  TVector3 H(H_temp[0], H_temp[1], H_temp[2]);
470  H = H*(10./tesla);
471 
472  double HA = H.Mag();
473  double HAM = HA*p_inv;
474  double HM;
475  if(HA < 1E-6)
476  {
477  HM = 0.;
478  }
479  else
480  {
481  HM = charge/HA;
482  }
483 
484  double Q = -HAM*c_light/(km/ns);
485  double sinz = -H.Dot(UN)*HM;
486  double cosz = H.Dot(VN)*HM;
487 
488  jac_sd2sc[1][3] = -Q*TVW[1]*sinz;
489  jac_sd2sc[1][4] = -Q*TVW[2]*sinz;
490  jac_sd2sc[2][3] = -Q*TVW[1]*cosz*cosl1;
491  jac_sd2sc[2][4] = -Q*TVW[2]*cosz*cosl1;
492  }
493 }

495 void TrackExtrapolator::TRSCSD(int charge, G4ThreeVector mom_input, G4ThreeVector pos_input)
496 {
498  TVector3 mom(mom_input[0], mom_input[1], mom_input[2]);
499  TVector3 pos(pos_input[0], pos_input[1], pos_input[2]);
500 
501  mom *= 0.001;
502  pos *= 0.1;
503 
505  TVector3 DJ(1., 0., 0.);
506  TVector3 DK(0., 1., 0.);
507  TVector3 DI = DJ.Cross(DK);
508 
510  double p_inv = 1./mom.Mag();
511  //double vp = mom.Dot(DJ)/mom.Dot(DI);
512  //double wp = mom.Dot(DK)/mom.Dot(DI);
513  double lambda = TMath::Pi()/2. - mom.Theta();
514  double phi = mom.Phi();
515 
516  double cosl = cos(lambda);
517  double sinp = sin(phi);
518  double cosp = cos(phi);
519 
520  TVector3 TN(cosl*cosp, cosl*sinp, sin(lambda));
521  TVector3 TVW;
522  TVW.SetX(TN.Dot(DI));
523  TVW.SetY(TN.Dot(DJ));
524  TVW.SetZ(TN.Dot(DK));
525 
526  double T1R = 1./TVW[0];
527  double T2R = T1R*T1R;
528  TVector3 UN(-sinp, cosp, 0.);
529  TVector3 VN(-TN.Z()*UN.Y(), TN.Z()*UN.X(), cosl);
530 
531  double UJ = UN.Dot(DJ);
532  double UK = UN.Dot(DK);
533  double VJ = VN.Dot(DJ);
534  double VK = VN.Dot(DK);
535  double UI = UN.Dot(DI);
536  double VI = VN.Dot(DI);
537 
538  jac_sc2sd.Zero();
539  jac_sc2sd[0][0] = 1.;
540  jac_sc2sd[1][1] = -UK*T2R;
541  jac_sc2sd[1][2] = VK*cosl*T2R;
542  jac_sc2sd[2][1] = UJ*T2R;
543  jac_sc2sd[2][2] = -VJ*cosl*T2R;
544  jac_sc2sd[3][3] = VK*T1R;
545  jac_sc2sd[3][4] = -UK*T1R;
546  jac_sc2sd[4][3] = -VJ*T1R;
547  jac_sc2sd[4][4] = UJ*T1R;
548 
549  const G4Field* field = G4TransportationManager::GetTransportationManager()->GetFieldManager()->GetDetectorField();
550  if(charge != 0 && field)
551  {
552  double pos_cm[3];
553  pos_cm[0] = pos.X()*cm;
554  pos_cm[1] = pos.Y()*cm;
555  pos_cm[2] = pos.Z()*cm;
556 
557  double H_temp[3];
558  field->GetFieldValue(pos_cm, H_temp);
559  TVector3 H(H_temp[0], H_temp[1], H_temp[2]);
560  H = H*(10./tesla);
561 
562  double HA = H.Mag();
563  double HAM = HA*p_inv;
564  double HM;
565  if(HA < 1E-6)
566  {
567  HM = 0.;
568  }
569  else
570  {
571  HM = charge/HA;
572  }
573 
574  double Q = -HAM*c_light/(km/ns);
575  double sinz = -H.Dot(UN)*HM;
576  double cosz = H.Dot(VN)*HM;
577  double T3R = Q*T1R*T1R*T1R;
578 
579  jac_sc2sd[1][3] = -UI*(VK*cosz - UK*sinz)*T3R;
580  jac_sc2sd[1][4] = -VI*(VK*cosz - UK*sinz)*T3R;
581  jac_sc2sd[2][3] = UI*(VJ*cosz - UJ*sinz)*T3R;
582  jac_sc2sd[2][4] = VI*(VJ*cosz - UJ*sinz)*T3R;
583  }

586 double TrackExtrapolator::extrapolateToIP()
587 {
588  //Store the steps on each point
589  G4ThreeVector mom[NSLICES_FMAG + NSTEPS_TARGET + 1];
590  G4ThreeVector pos[NSLICES_FMAG + NSTEPS_TARGET + 1];
591 
592  //Step size in FMAG/target area, unit is cm.
593  double step_fmag = FMAG_LENGTH/NSLICES_FMAG*cm;
594  double step_target = fabs(Z_UPSTREAM)/NSTEPS_TARGET*cm;
595 
596  //Start from FMAG face downstream
597  extrapolateTo(FMAG_LENGTH);
598  pos[0] = pos_f;
599  mom[0] = mom_f;
600 
601  //Now make the real swimming
602  int iStep = 1;
603  for(; iStep <= NSLICES_FMAG; ++iStep)
604  {
605  pos_i = pos[iStep-1];
606  mom_i = mom[iStep-1];
607 
608  extrapolateTo((pos_i[2] - step_fmag)*mm/cm);
609 
610  pos[iStep] = pos_f;
611  mom[iStep] = mom_f;
612  }
613 
614  for(; iStep < NSLICES_FMAG+NSTEPS_TARGET+1; ++iStep)
615  {
616  pos_i = pos[iStep-1];
617  mom_i = mom[iStep-1];
618 
619  extrapolateTo((pos_i[2] - step_target)*mm/cm);
620 
621  pos[iStep] = pos_f;
622  mom[iStep] = mom_f;
623  }
624 
625  //Find the one step with minimum DCA
626  double dca_min = 1E6;
627  for(int i = 0; i < NSLICES_FMAG+NSTEPS_TARGET+1; ++i)
628  {
629  double dca = sqrt(pos[i][0]*pos[i][0] + pos[i][1]*pos[i][1]);
630  if(dca < dca_min)
631  {
632  dca_min = dca;
633  iStep = i;
634  }
635  }
636 
637  return pos[iStep][2]*mm/cm;
638 }
