285 float PHG4HcalSteppingAction::GetLightCorrection(float r) {
286  float m = (light_balance_outer_corr_ - light_balance_inner_corr_)/(light_balance_outer_radius_ - light_balance_inner_radius_);
287  float b = light_balance_inner_corr_ - m*light_balance_inner_radius_;
288  float value = m*r+b;
289  if (value > 1.0) return 1.0;
290  if (value < 0.0) return 0.0;
