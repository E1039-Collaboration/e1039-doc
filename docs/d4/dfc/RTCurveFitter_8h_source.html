66 inline void AnaFit::CalcChi2(const std::vector<double>& pars, double& chi2, int& ndf) const
67 {
68  double t1 = pars[0];
69  double t0 = pars[1];
70  double dr = pars[2];
71  TSpline3* spl = CreateSpline(pars);
72 
73  chi2 = 0;
74  ndf = 0;
75  double r_width = m_h2_RT->GetYaxis()->GetBinWidth(1);
76  for (int it = m_h2_RT->GetNbinsX(); it > 0; it--) {
77  double cont_t = m_h2_RT->Integral(it, it);
78  if (cont_t < 10) continue;
79  double t_cent = m_h2_RT->GetXaxis()->GetBinCenter(it);
80  double r_func = EvalR(t_cent, t1, t0, spl);
81  for (int ir = m_h2_RT->GetNbinsY(); ir > 0; ir--) {
82  double cont = m_h2_RT->GetBinContent(it, ir);
83  if (cont <= 0) continue;
84  double r_cent = m_h2_RT->GetYaxis()->GetBinCenter(ir);
85  double cont_func = cont_t * r_width * exp( -pow( (r_cent-r_func)/dr, 2 )/2 ) / (sqrt(2*TMath::Pi()) * dr);
86  double chi2p = pow(cont - cont_func, 2) / cont;
87  //std::cout << "XXX " << it << " " << ir << " : " << dr << " " << cont_t << " " << cont_func << " " << cont << " " << chi2p << std::endl;
88  chi2 += chi2p;
89  ndf++;
90  }
91  }
92  ndf -= m_minuit->GetNumberFreeParameters();
93  delete spl;
94 }
