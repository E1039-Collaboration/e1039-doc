319  string line_mapping;
320  while ( getline( istream_mapping, line_mapping ) )
321  {
322  /* Skip lines starting with / including a '#' */
323  if ( line_mapping.find("#") != string::npos )
324  {

328  }

330  }
331 
332  istringstream iss(line_mapping);
333 
334  /* If line starts with keyword Tower, add to tower positions */
335  if ( line_mapping.find("Tower ") != string::npos )
336  {
337  unsigned idx_j, idx_k, idx_l;
338  G4double pos_x, pos_y, pos_z;
339  G4double size_x, size_y, size_z;
340  G4double rot_x, rot_y, rot_z;
341  G4double dummy;
342  string dummys;
343 
344  /* read string- break if error */
345  if ( !( iss >> dummys >> dummy >> idx_j >> idx_k >> idx_l >> pos_x >> pos_y >> pos_z >> size_x >> size_y >> size_z >> rot_x >> rot_y >> rot_z ) )
346  {
347  cerr << "ERROR in PHG4ForwardHcalDetector: Failed to read line in mapping file " << _mapping_tower_file << endl;
348  exit(1);

350 
351  /* Construct unique name for tower */
352  /* Mapping file uses cm, this class uses mm for length */

356 
357  /* Add Geant4 units */

361

369  }

372  /* If this line is not a comment and not a tower, save parameter as string / value. */
373  string parname;

375 
376  /* read string- break if error */

380  exit(1);

385  }
