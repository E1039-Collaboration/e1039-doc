284 void SRecTrack::swimToVertex(TVector3* pos, TVector3* mom)
285 {
286  //Store the steps on each point (center of the interval)
287  bool cleanupPos = false;
288  bool cleanupMom = false;
289  if(pos == NULL)
290  {
291  pos = new TVector3[NSLICES_FMAG + NSTEPS_TARGET + 1];
292  cleanupPos = true;
293  }
294  if(mom == NULL)
295  {
296  mom = new TVector3[NSLICES_FMAG + NSTEPS_TARGET + 1];
297  cleanupMom = true;
298  }
299 
300  //E-loss and pT-kick per length, note the eloss is done in half-slices
301  double eloss_unit_0 = ELOSS_FMAG_P0/FMAG_LENGTH;
302  double eloss_unit_1 = ELOSS_FMAG_P1/FMAG_LENGTH;
303  double eloss_unit_2 = ELOSS_FMAG_P2/FMAG_LENGTH;
304  double eloss_unit_3 = ELOSS_FMAG_P3/FMAG_LENGTH;
305  double eloss_unit_4 = ELOSS_FMAG_P4/FMAG_LENGTH;
306  double ptkick_unit = PT_KICK_FMAG/FMAG_LENGTH;
307 
308  //Step size in FMAG/target area
309  double step_fmag = FMAG_LENGTH/NSLICES_FMAG/2.; //note that in FMag, each step is devided into two slices
310  double step_target = fabs(Z_UPSTREAM)/NSTEPS_TARGET;
311 
312  //track slope/location in upstream
313  double tx = fState.front()[1][0];
314  double ty = fState.front()[2][0];
315  double x0 = fState.front()[3][0];
316  double y0 = fState.front()[4][0];
317  double z0 = fZ.front();
318 
319  //Initial position should be on the downstream face of beam dump
320  pos[0].SetXYZ(x0 + tx*(FMAG_LENGTH - z0), y0 + ty*(FMAG_LENGTH - z0), FMAG_LENGTH);
321  mom[0] = getMomentumVecSt1();
322 
323  //Charge of the track
324  double charge = getCharge();
325 
326  //Now make the swim
327  int iStep = 1;
328  for(; iStep <= NSLICES_FMAG; ++iStep)
329  {
330  //Make pT kick at the center of slice, add energy loss at both first and last half-slice
331  //Note that ty is the global class data member, which does not change during the entire swimming
332  double tx_i = mom[iStep-1].Px()/mom[iStep-1].Pz();
333  double tx_f = tx_i + 2.*charge*ptkick_unit*step_fmag/sqrt(mom[iStep-1].Px()*mom[iStep-1].Px() + mom[iStep-1].Pz()*mom[iStep-1].Pz());
334 
335  TVector3 trajVec1(tx_i*step_fmag, ty*step_fmag, step_fmag);
336  TVector3 pos_b = pos[iStep-1] - trajVec1;
337 
338  double p_tot_i = mom[iStep-1].Mag();
339  double p_tot_b;
340  if(pos_b[2] > FMAG_HOLE_LENGTH || pos_b.Perp() > FMAG_HOLE_RADIUS)
341  {
342  p_tot_b = p_tot_i + (eloss_unit_0 + p_tot_i*eloss_unit_1 + p_tot_i*p_tot_i*eloss_unit_2 + p_tot_i*p_tot_i*p_tot_i*eloss_unit_3 + p_tot_i*p_tot_i*p_tot_i*p_tot_i*eloss_unit_4)*trajVec1.Mag();
343  }
344  else
345  {
346  p_tot_b = p_tot_i;
347  }
348 
349  TVector3 trajVec2(tx_f*step_fmag, ty*step_fmag, step_fmag);
350  pos[iStep] = pos_b - trajVec2;
351 
352  double p_tot_f;
353  if(pos[iStep][2] > FMAG_HOLE_LENGTH || pos[iStep].Perp() > FMAG_HOLE_RADIUS)
354  {
355  p_tot_f = p_tot_b + (eloss_unit_0 + p_tot_b*eloss_unit_1 + p_tot_b*p_tot_b*eloss_unit_2 + p_tot_b*p_tot_b*p_tot_b*eloss_unit_3 + p_tot_b*p_tot_b*p_tot_b*p_tot_b*eloss_unit_4)*trajVec2.Mag();
356  }
357  else
358  {
359  p_tot_f = p_tot_b;
360  }
361 
362  //Now the final position and momentum in this step
363  double pz_f = p_tot_f/sqrt(1. + tx_f*tx_f + ty*ty);
364  mom[iStep].SetXYZ(pz_f*tx_f, pz_f*ty, pz_f);
365  pos[iStep] = pos[iStep-1] - trajVec1 - trajVec2;
366 
367  //Save the dump position when applicable
368  if(fabs(pos_b.Z() - Z_DUMP) < step_fmag)
369  {
370  double dz = Z_DUMP - pos_b.Z();
371  if(dz < 0)
372  {
373  setDumpPos(pos_b + TVector3(tx_f*dz, ty*dz, dz));
374  setDumpMom(mom[iStep]);
375  }
376  else
377  {
378  setDumpPos(pos_b + TVector3(tx_i*dz, ty*dz, dz));
379  setDumpMom(mom[iStep-1]);
380  }
381  }
382 
383 #ifdef _DEBUG_ON_LEVEL_2
384  std::cout << "FMAG: " << iStep << ": " << pos[iStep-1][2] << " ==================>>> " << pos[iStep][2] << std::endl;
385  std::cout << mom[iStep-1][0]/mom[iStep-1][2] << " " << mom[iStep-1][1]/mom[iStep-1][2] << " " << mom[iStep-1][2] << " ";
386  std::cout << pos[iStep-1][0] << " " << pos[iStep-1][1] << " " << pos[iStep-1][2] << std::endl << std::endl;
387  std::cout << mom[iStep][0]/mom[iStep][2] << " " << mom[iStep][1]/mom[iStep][2] << " " << mom[iStep][2] << " ";
388  std::cout << pos[iStep][0] << " " << pos[iStep][1] << " " << pos[iStep][2] << std::endl << std::endl;
389 #endif
390  }
391 
392  for(; iStep < NSLICES_FMAG+NSTEPS_TARGET+1; ++iStep)
393  {
394  //Simple straight line flight
395  double tx_i = mom[iStep-1].Px()/mom[iStep-1].Pz();
396  TVector3 trajVec(tx_i*step_target, ty*step_target, step_target);
397 
398  mom[iStep] = mom[iStep-1];
399  pos[iStep] = pos[iStep-1] - trajVec;
400 
401 #ifdef _DEBUG_ON_LEVEL_2
402  std::cout << "TARGET: " << iStep << ": " << pos[iStep-1][2] << " ==================>>> " << pos[iStep][2] << std::endl;
403  std::cout << mom[iStep-1][0]/mom[iStep-1][2] << " " << mom[iStep-1][1]/mom[iStep-1][2] << " " << mom[iStep-1][2] << " ";
404  std::cout << pos[iStep-1][0] << " " << pos[iStep-1][1] << " " << pos[iStep-1][2] << std::endl << std::endl;
405  std::cout << mom[iStep][0]/mom[iStep][2] << " " << mom[iStep][1]/mom[iStep][2] << " " << mom[iStep][2] << " ";
406  std::cout << pos[iStep][0] << " " << pos[iStep][1] << " " << pos[iStep][2] << std::endl << std::endl;
407 #endif
408  }
409 
410  //Now the swimming is done, find the point with closest distance of approach, let iStep store the index of that step
411  double dca_min = 1E9;
412  double dca_xmin = 1E9;
413  double dca_ymin = 1E9;
414 
415  iStep = NSLICES_FMAG+NSTEPS_TARGET; // set the default point to the most upstream
416  int iStep_x = iStep; // the point when track cross beam line in X and in Y
417  int iStep_y = iStep; // both intialized with the most upstream position
418  for(int i = 0; i < NSLICES_FMAG+NSTEPS_TARGET+1; ++i)
419  {
420  if(FMAGSTR*charge*mom[i].Px() < 0.) continue; // this is the upstream accidental cross, ignore
421 
422  double dca = (pos[i] - TVector3(X_VTX, Y_VTX, pos[i].Z())).Perp();
423  if(dca < dca_min)
424  {
425  dca_min = dca;
426  iStep = i;
427  }
428 
429  double dca_x = fabs(pos[i].X() - X_VTX);
430  if(dca_x < dca_xmin)
431  {
432  dca_xmin = dca_x;
433  iStep_x = i;
434  }
435 
436  double dca_y = fabs(pos[i].Y() - Y_VTX);
437  if(dca_y < dca_ymin)
438  {
439  dca_ymin = dca_y;
440  iStep_y = i;
441  }
442  }
443 
444  setVertexFast(mom[iStep], pos[iStep]);
445  setDumpFacePos(pos[NSLICES_FMAG]);
446  setDumpFaceMom(mom[NSLICES_FMAG]);
447  setTargetPos(fDumpFacePos + TVector3(fDumpFaceMom.Px()/fDumpFaceMom.Pz()*Z_TARGET, fDumpFaceMom.Py()/fDumpFaceMom.Pz()*Z_TARGET, Z_TARGET));
448  setTargetMom(mom[NSLICES_FMAG]);
449 
450  double dz_x = -pos[iStep_x].X()/mom[iStep_x].Px()*mom[iStep_x].Pz();
451  setXVertexPos(pos[iStep_x] + TVector3(mom[iStep_x].Px()/mom[iStep_x].Pz()*dz_x, mom[iStep_x].Py()/mom[iStep_x].Pz()*dz_x, dz_x));
452  setXVertexMom(mom[iStep_x]);
453 
454  double dz_y = -pos[iStep_y].Y()/mom[iStep_y].Py()*mom[iStep_y].Pz();
455  setYVertexPos(pos[iStep_y] + TVector3(mom[iStep_y].Px()/mom[iStep_y].Pz()*dz_y, mom[iStep_y].Py()/mom[iStep_y].Pz()*dz_y, dz_y));
456  setYVertexMom(mom[iStep_y]);
457 
458 #ifdef _DEBUG_ON_LEVEL_2
459  std::cout << "The one with minimum DCA is: " << iStep << ": " << std::endl;
460  std::cout << mom[iStep][0]/mom[iStep][2] << " " << mom[iStep][1]/mom[iStep][2] << " " << mom[iStep][2] << " ";
461  std::cout << pos[iStep][0] << " " << pos[iStep][1] << " " << pos[iStep][2] << std::endl << std::endl;
462 #endif
463 
464 #ifdef _ENABLE_KF
465  updateVtxHypothesis();
466 #endif
467 
468  if(cleanupPos) delete[] pos;
469  if(cleanupMom) delete[] mom;
470 }

487 void SRecDimuon::calcVariables()
488 {
489  Double_t mp = 0.938;
490  Double_t ebeam = 120.;
491 
492  TLorentzVector p_beam(0., 0., sqrt(ebeam*ebeam - mp*mp), ebeam);
493  TLorentzVector p_target(0., 0., 0., mp);
494 
495  TLorentzVector p_cms = p_beam + p_target;
496  TLorentzVector p_sum = p_pos + p_neg;
497 
498  mass = p_sum.M();
499  pT = p_sum.Perp();
500 
501  x1 = (p_target*p_sum)/(p_target*p_cms);
502  x2 = (p_beam*p_sum)/(p_beam*p_cms);
503 
504  Double_t s = p_cms.M2();
505  TVector3 bv_cms = p_cms.BoostVector();
506  p_sum.Boost(-bv_cms);
507  xF = 2.*p_sum.Pz()/TMath::Sqrt(s)/(1. - mass*mass/s);
508 
509  costh = 2.*(p_neg.E()*p_pos.Pz() - p_pos.E()*p_neg.Pz())/mass/sqrt(mass*mass + pT*pT);
510  phi = atan2(2.*sqrt(mass*mass + pT*pT)*(p_neg.X()*p_pos.Y() - p_pos.X()*p_neg.Y()), mass*(p_pos.X()*p_pos.X() - p_neg.X()*p_neg.X() + p_pos.Y()*p_pos.Y() - p_neg.Y()*p_neg.Y()));
511  mass_single = (p_pos_single + p_neg_single).M();
512 }

514 bool SRecDimuon::isValid()
515 {
516  //Chisq of vertex fit
517  if(chisq_kf > 15. || chisq_kf < 0.) return false;
518 
519  //Kinematic cuts
520  if(FMAGSTR*(p_pos.Px() - p_neg.Px()) < 0.) return false;
521  if(fabs(xF) > 1.) return false;
522  if(x1 < 0. || x1 > 1.) return false;
523  if(x2 < 0. || x2 > 1.) return false;
524  if(mass < 0. || mass > 10.) return false;
525  if(fabs(vtx.X()) > 2.) return false;
526  if(fabs(vtx.Y()) > 2.) return false;
