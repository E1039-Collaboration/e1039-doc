321 std::pair<int, int>
322 PHG4CylinderGeom_Spacalv3::get_tower_z_phi_ID(const int tower_ID,
323  const int sector_ID) const
324 {
325  // tower_ID to eta/z within a sector
326  int z_bin = floor(tower_ID / 10);
327 
328  int phi_bin_in_sec = -1;
329 
330  if (get_config() == kFullProjective_2DTaper_SameLengthFiberPerTower or get_config() == kFullProjective_2DTaper)
331  // colume ID is from -x to +x at the top of the detector, which is reverse of the phi bin direction.
332  phi_bin_in_sec = max_phi_bin_in_sec - (tower_ID % 10);
333  else if (get_config() == kFullProjective_2DTaper_Tilted_SameLengthFiberPerTower or get_config() == kFullProjective_2DTaper_Tilted)
334  phi_bin_in_sec = (tower_ID % 10);
335 
336  if (!(phi_bin_in_sec < max_phi_bin_in_sec and phi_bin_in_sec >= 0))
337  {
338  cout
339  << "PHG4CylinderGeom_Spacalv3::get_tower_z_phi_ID - Fatal Error - invalid in put with "
340  << "tower_ID = " << tower_ID << ", sector_ID = " << sector_ID<< ", phi_bin_in_sec = " << phi_bin_in_sec
341  << ". Dump object:" << endl;
342  Print();
343  }
344 
345  assert(phi_bin_in_sec < max_phi_bin_in_sec and phi_bin_in_sec >= 0);
346 
347  int phi_bin = sector_ID * max_phi_bin_in_sec + phi_bin_in_sec;
348 
349  return make_pair(z_bin, phi_bin);
350 }

353 double PHG4CylinderGeom_Spacalv3::
354 get_tower_radial_position(const PHG4CylinderGeom_Spacalv3::geom_tower & tower) const
355 {
356  if (get_config() == kFullProjective_2DTaper_SameLengthFiberPerTower or get_config() == kFullProjective_2DTaper)
357  return tower.centralY;
358  else if (get_config() == kFullProjective_2DTaper_Tilted_SameLengthFiberPerTower or get_config() == kFullProjective_2DTaper_Tilted)
359  {
360  const double outter_wall_shift = get_sidewall_thickness() + get_sidewall_outer_torr() + get_assembly_spacing();
361  assert(outter_wall_shift>=0);
362  const double tilted_radial_shift = outter_wall_shift / sin(M_PI / get_azimuthal_n_sec());
363  assert(tilted_radial_shift>=0);
364  const double tower_radial = //
365  tilted_radial_shift * cos(get_azimuthal_tilt()) +
366  get_half_radius() * sin(get_azimuthal_tilt()) * sin(get_azimuthal_tilt()) +
367  tower.centralY * cos(get_azimuthal_tilt());
368 
369 
370  if (get_construction_verbose()>=2)
371  {
372  cout
373  << "PHG4CylinderGeom_Spacalv3::get_tower_radial_position - tower radial adjustment: "
374  "from "<<tower.centralY<<" to "<<tower_radial
375  << endl;
376  }
377 
378  return tower_radial;
379  }
380  else
381  {
382  cout
383  << "PHG4CylinderGeom_Spacalv3::get_tower_radial_position - ERROR - "
384  "unsupported configuration!"
385  << endl;
386  Print();
387  exit(10);
388  }
389  return NAN;
390 }

740 PHG4CylinderGeom_Spacalv3::load_demo_sector_tower_map4()
741 {
742  cout << "PHG4CylinderGeom_Spacalv3::load_demo_sector_tower_map4 - "
743  << "FermiLab test beam 2014. Need to move to calibration database"
744  << endl;
745 
746  // From Oleg's documents
747 
748  const int ny = 3;
749  const int nx = 3;
750 
751  const double inch_to_cm = 2.54;
752 
753  const double screen_size_y = 2.108 * inch_to_cm;
754  const double screen_size_x = 1.049 * inch_to_cm;
755 
756  const double z_screen_6_7 = 0.5 * (6.6 + 6.75) * inch_to_cm;
757  const double angle_screen_6_7 = 64.78 / 180. * M_PI;
758  const double nawrrow_width_x = screen_size_x * sin(angle_screen_6_7);
759 
760  const double z_screen_1_2 = 0.5 * (1.35 + 1.6) * inch_to_cm;
761  const double angle_screen_1_2 = 90.56 / 180. * M_PI;
762  const double wide_width_x = screen_size_x * sin(angle_screen_1_2);
763 
764  const double module_length = z_screen_6_7 - z_screen_1_2;
765  assert(module_length > 0);
766 
767  //tapering, dxwidth/dlength
768  const double tapering_ratio = (wide_width_x - nawrrow_width_x)
769  / module_length;
770  assert(tapering_ratio < 1);
771  assert(tapering_ratio > 0);
772 
773  fiber_clading_thickness = 0.003;
774  fiber_core_diameter = 0.050 - fiber_clading_thickness * 2;
775 
776  assembly_spacing = 0.002500;
777 
778  radius = (nawrrow_width_x) / tapering_ratio;
779  thickness = module_length * 1.5; // keep a large torlerence space
780  zmin = -(0.5 * ny * screen_size_y + 2 * assembly_spacing * (ny + 1));
781  zmax = -zmin;
782  azimuthal_n_sec = floor(2 * M_PI / atan(tapering_ratio));
783  max_phi_bin_in_sec = 1;
784 
785  const double nawrrow_width_x_construction = radius * 2
786  * tan(M_PI / azimuthal_n_sec) - 2 * assembly_spacing;
787  const double wide_width_x_construction = (radius + module_length) * 2
788  * tan(M_PI / azimuthal_n_sec) - 2 * assembly_spacing;
789 
790  cout << "PHG4CylinderGeom_Spacalv3::load_demo_sector_tower_map4 - "
791 
792  << "Adjust wide end width by ratio of "
793  << wide_width_x_construction / wide_width_x
794  << " and narrow end by ratio of "
795  << nawrrow_width_x_construction / nawrrow_width_x << endl;
796 
797  sector_map.clear();
798  for (int sec = 0; sec < nx; ++sec)
799  {
800  const double rot = twopi / (double) (get_azimuthal_n_sec())
801  * ((double) (sec) - 1);
802 
803  sector_map[sec] = rot;
804  }
805 
806  azimuthal_tilt = 0;
807  azimuthal_seg_visible = true;
808  polar_taper_ratio = 1.;
809  sidewall_thickness = 0;
810  sidewall_outer_torr = 0;
811  sector_tower_map.clear();
812 
813  for (int y = 0; y < ny; y++)
814  {
815 
816  geom_tower geom;
817  geom.id = y;
818  geom.pDz = module_length / 2;
819  geom.pTheta = 0.;
820  geom.pPhi = 0.;
821  geom.pAlp1 = 0.;
822  geom.pAlp2 = 0.;
823  geom.pDy1 = 0.5 * screen_size_y;
824  geom.pDx1 = 0.5 * nawrrow_width_x_construction;
825  geom.pDx2 = 0.5 * nawrrow_width_x_construction;
826  geom.pDy2 = 0.5 * screen_size_y;
827  geom.pDx3 = 0.5 * wide_width_x_construction;
828  geom.pDx4 = 0.5 * wide_width_x_construction;
829 
830  geom.centralX = 0.;
831  geom.centralY = module_length * 0.5 + radius + assembly_spacing;
832  geom.centralZ = screen_size_y * (y - 1);
833 
834  geom.pRotationAngleX = -M_PI / 2.;
835  geom.ModuleSkinThickness = 0.010000;
836  geom.NFiberX = 30;
837  geom.NFiberY = 26 * 2 * 2;
838  sector_tower_map[geom.id] = geom;
839  }
