1131 * matrix and the inverse.

1157  END DO

1165  jj=jj+j

1167  IF(abs(v(jj)).GT.max(abs(vkk),eps*diag(j))) THEN
1168  vkk=v(jj) ! pivot (candidate)
1169  k =j ! index of pivot

1171  END IF
1172  END IF

1185  jl=jl+j
1186  ELSE

1191  END IF

1197  jl=jl+1
1198  IF(l.EQ.k) THEN

1200  ELSE
1201  IF(l.LT.k) THEN
1202  lk=lk+1
1203  ELSE
1204  lk=lk+l-1
1205  END IF

1207  END IF
