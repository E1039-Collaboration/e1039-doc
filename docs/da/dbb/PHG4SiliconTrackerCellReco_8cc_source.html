244 int PHG4SiliconTrackerCellReco::CheckEnergy(PHCompositeNode *topNode)

250 
251  PHG4HitContainer::ConstRange hit_begin_end = g4hit->getHits();
252  PHG4HitContainer::ConstIterator hiter;
253  for (hiter = hit_begin_end.first; hiter != hit_begin_end.second; ++hiter)
254  sum_energy_g4hit += hiter->second->get_edep();
255 
256  PHG4CellContainer::ConstRange cell_begin_end = cells->getCells();
257  PHG4CellContainer::ConstIterator citer;
258  for (citer = cell_begin_end.first; citer != cell_begin_end.second; ++citer)
259  sum_energy_cells += citer->second->get_edep();
260 
261  // the fractional eloss for particles traversing eta bins leads to minute rounding errors
262  if (fabs(sum_energy_cells - sum_energy_g4hit) / sum_energy_g4hit > 1e-6)
263  {
264  std::cout << "energy mismatch between cells: " << sum_energy_cells
265  << " and hits: " << sum_energy_g4hit
266  << " diff sum(cells) - sum(hits): " << sum_energy_cells - sum_energy_g4hit
267  << std::endl;
268 
269  return -1;
270  }
271  else
272  {
273  if (verbosity > 0)
274  std::cout << Name() << ": total energy for this event: " << sum_energy_g4hit << " GeV" << std::endl;
275  }
