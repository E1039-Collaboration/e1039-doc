398 int
399 PHG4Prototype2InnerHcalDetector::ConstructInnerHcal(G4LogicalVolume* hcalenvelope)
400 {
401  G4LogicalVolume* steel_plate = ConstructSteelPlate(hcalenvelope); // bottom steel plate
402  G4LogicalVolume* scintibox = NULL;
403  if (params->get_int_param("hi_eta"))
404  {
405  scintibox = ConstructScintillatorBoxHiEta(hcalenvelope);
406  }
407  else
408  {
409  scintibox = ConstructScintillatorBox(hcalenvelope);
410  }
411  double phi = 0.;
412  double phislat = 0.;
413  ostringstream name;
414  // the coordinate of the center of the bottom of the bottom steel plate
415  // to get the radius of the circle which is the center of the scintillator box
416  double bottom_xmiddle_steel_tile = (steel_plate_corner_lower_right.x()+steel_plate_corner_lower_left.x())/2.;
417  double bottom_ymiddle_steel_tile = (steel_plate_corner_lower_left.y()+steel_plate_corner_lower_right.y())/2.;
418  double middlerad = sqrt(bottom_xmiddle_steel_tile*bottom_xmiddle_steel_tile + bottom_ymiddle_steel_tile * bottom_ymiddle_steel_tile);
419  double philow = atan((bottom_ymiddle_steel_tile-scinti_gap/2.-0.87*mm)/bottom_xmiddle_steel_tile);
420  double scintiangle = GetScintiAngle();
421  for (int i = 0; i < n_steel_plates; i++)
422  // for (int i = 0; i < 2; i++)
423  {
424  name.str("");
425  name << "InnerHcalSteel_" << i;
426  G4RotationMatrix *Rot = new G4RotationMatrix();
427  Rot->rotateZ(phi*rad);
428  G4ThreeVector g4vec(0,0,0);
429  innerhcalassembly->AddPlacedVolume(steel_plate,g4vec,Rot);
430  if (i > 0)
431  {
432  double ypos = sin(phi+philow) * middlerad;
433  double xpos = cos(phi+philow) * middlerad;
434  name.str("");
435  name << "InnerHcalScintiBox_" << i;
436  Rot = new G4RotationMatrix();
437  Rot->rotateZ(scintiangle+phislat);
438  G4ThreeVector g4vecsc(xpos, ypos, 0);
439  innerhcalassembly->AddPlacedVolume(scintibox,g4vecsc,Rot);
440  phislat += deltaphi;
441  }
442  phi += deltaphi;
443  }
444  return 0;
445 }
