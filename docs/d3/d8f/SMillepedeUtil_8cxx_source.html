127 void MPNode::setDerivatives(double z, double cosphi, double sinphi, double tx, double ty, double x0, double y0)
128 {
129  //Global derivatives
130  dwdz = (tx*cosphi + ty*sinphi);
131  dwdphi = (-x0*sinphi - tx*z*sinphi + y0*cosphi + ty*z*cosphi);
132  dwdw = -1.;
133 
134  //Local derivatives
135  dwdx = cosphi;
136  dwdy = sinphi;
137  dwdtx = z*cosphi;
138  dwdty = z*sinphi;
139 }
