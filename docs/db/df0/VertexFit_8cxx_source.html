214 double VertexFit::findDimuonVertexFast(SRecTrack& track1, SRecTrack& track2)
215 {
216  //Swim both tracks all the way down, and store the numbers
217  TVector3 pos1[NSLICES_FMAG + NSTEPS_TARGET + 1];
218  TVector3 pos2[NSLICES_FMAG + NSTEPS_TARGET + 1];
219  TVector3 mom1[NSLICES_FMAG + NSTEPS_TARGET + 1];
220  TVector3 mom2[NSLICES_FMAG + NSTEPS_TARGET + 1];
221  track1.swimToVertex(pos1, mom1);
222  track2.swimToVertex(pos2, mom2);
223 
224  int iStep_min = -1;
225  double dist_min = 1E6;
226  int charge1 = track1.getCharge();
227  int charge2 = track2.getCharge();
228  for(int iStep = 0; iStep < NSLICES_FMAG + NSTEPS_TARGET + 1; ++iStep)
229  {
230  double dist = (pos1[iStep] - pos2[iStep]).Perp();
231  if(dist < dist_min && FMAGSTR*charge1*mom1[iStep].Px() > 0 && FMAGSTR*charge2*mom2[iStep].Px() > 0)
232  {
233  iStep_min = iStep;
234  dist_min = dist;
235  }
236  }
237 
238  if(iStep_min == -1) return Z_DUMP;
239  return pos1[iStep_min].Z();
240 }

347 int VertexFit::findVertex()
348 {
349  int nIter = 0;
350  double _chisq_kalman_prev = 1E6;
351  for(; nIter < _max_iteration; ++nIter)
352  {
353 #ifdef _DEBUG_ON
354  LogInfo("Iteration: " << nIter);
355 #endif
356 
357  _chisq_vertex = 0.;
358  _chisq_kalman = 0.;
359  for(unsigned int j = 0; j < _trkpar_curr.size(); j++)
360  {
361  _node_vertex.resetFlags();
362  //_node_vertex.getMeasurement() = _vtxpar_curr._r.GetSub(0, 1, 0, 0);
363  //_node_vertex.getMeasurementCov() = _vtxpar_curr._cov.GetSub(0, 1, 0, 1);
364  _node_vertex.setZ(_vtxpar_curr._r[2][0]);
365 
366  _kmfit->setCurrTrkpar(_trkpar_curr[j]);
367  if(!_kmfit->fit_node(_node_vertex))
368  {
369 #ifdef _DEBUG_ON
370  LogInfo("Vertex fit for this track failed!");
371 #endif
372  _chisq_kalman = 1E5;
373  break;
374  }
375  else
376  {
377  _chisq_kalman += _node_vertex.getChisq();
378  }
379 
380  updateVertex();
381  }
382 
384 #ifdef _DEBUG_ON
385  LogInfo("At this iteration: ");
386  LogInfo(_vtxpar_curr._r[2][0] << " === " << _node_vertex.getZ() << " : " << _chisq_kalman << " === " << _chisq_vertex << " : " << _vtxpar_curr._r[0][0] << " === " << _vtxpar_curr._r[1][0]);
387 #endif
388  if(_vtxpar_curr._r[2][0] < Z_UPSTREAM || _vtxpar_curr._r[2][0] > Z_DOWNSTREAM)
389  {
390  _chisq_kalman = 1E5;
391  _chisq_vertex = 1E5;
392  break;
393  }
394 
395  if(nIter > 0 && (fabs(_vtxpar_curr._r[2][0] - _node_vertex.getZ()) < _tolerance || _chisq_kalman > _chisq_kalman_prev)) break;
396  _chisq_kalman_prev = _chisq_kalman;
397  }
398 
399  return nIter+1;
400 }

402 void VertexFit::updateVertex()
403 {
404  double p = fabs(1./_node_vertex.getFiltered()._state_kf[0][0]);
405  double px = _node_vertex.getFiltered()._state_kf[1][0];
406  double py = _node_vertex.getFiltered()._state_kf[2][0];
407  double pz = sqrt(p*p - px*px - py*py);
408 
410  TMatrixD H(2, 3);
411  H.Zero();
412 
413  H[0][0] = 1.;
414  H[1][1] = 1.;
415  H[0][2] = -px/pz;
416  H[1][2] = -py/pz;
417 
418  TMatrixD vertex_dummy(3, 1);
419  vertex_dummy.Zero();
420  vertex_dummy[2][0] = _vtxpar_curr._r[2][0];
421 
422  TMatrixD mxy = _node_vertex.getFiltered()._state_kf.GetSub(3, 4, 0, 0);
423  TMatrixD Vxy = _node_vertex.getFiltered()._covar_kf.GetSub(3, 4, 3, 4);
424  TMatrixD S = SMatrix::invertMatrix(Vxy + SMatrix::getABCt(H, _vtxpar_curr._cov, H));
425  TMatrixD K = SMatrix::getABtC(_vtxpar_curr._cov, H, S);
426  TMatrixD zeta = mxy - H*(_vtxpar_curr._r - vertex_dummy);
427  TMatrixD _r_filtered = _vtxpar_curr._r + K*zeta;
428  TMatrixD _cov_filtered = _vtxpar_curr._cov - K*H*(_vtxpar_curr._cov);
429 
430  _chisq_vertex += SMatrix::getAtBC(zeta, S, zeta)[0][0];
431 
432  _vtxpar_curr._r = _r_filtered;
433  _vtxpar_curr._cov = _cov_filtered;
434 }

499 void VertexFit::fillEvaluation()
500 {
501  if(evalTree == NULL) return;
502 
503  choice_by_kf_eval = -1;
504  choice_by_vx_eval = -1;
505  double chisq_kf_min = 1E6;
506  double chisq_vx_min = 1E6;
507 
508  m_chisq_kf_eval = 0.;
509  m_z_vertex_eval = 0.;
510  for(int i = 0; i < nStart; ++i)
511  {
512  chisq_kf_eval[i] = chisq_km[i];
513  chisq_vx_eval[i] = chisq_vx[i];
514  z_vertex_eval[i] = z_vertex[i];
515  r_vertex_eval[i] = r_vertex[i];
516  z_start_eval[i] = z_start[i];
517 
518  m_chisq_kf_eval += chisq_kf_eval[i];
519  m_z_vertex_eval += z_vertex_eval[i];
520 
521  if(chisq_kf_min > chisq_km[i])
522  {
523  choice_by_kf_eval = i;
524  chisq_kf_min = chisq_km[i];
525  }
526 
527  if(chisq_vx_min > chisq_vx[i])
528  {
529  choice_by_vx_eval = i;
530  chisq_vx_min = chisq_vx[i];
531  }
532  }
533  m_chisq_kf_eval /= nStart;
534  m_z_vertex_eval /= nStart;
535 
536  s_chisq_kf_eval = 0.;
537  s_z_vertex_eval = 0.;
538  if(nStart == 1)
539  {
540  evalTree->Fill();
541  return;
542  }
543 
544  for(int i = 0; i < nStart; ++i)
545  {
546  s_chisq_kf_eval += ((m_chisq_kf_eval - chisq_kf_eval[i])*(m_chisq_kf_eval - chisq_kf_eval[i]));
547  s_z_vertex_eval += ((m_z_vertex_eval - z_vertex_eval[i])*(m_z_vertex_eval - z_vertex_eval[i]));
548  }
549  s_chisq_kf_eval = sqrt(s_chisq_kf_eval/(nStart-1));
550  s_z_vertex_eval = sqrt(s_z_vertex_eval/(nStart-1));
551 
552  evalTree->Fill();
553 }
