172 void
173 PHG4HcalSubsystem::SetTiltViaNcross(const int ncross)
174 {
175  if (ncross == 0)
176  {
177  cout << Name() << " invalid crossing number " << ncross
178  << " how do you think we can construct a meaningful detector with this number????" << endl;
179  exit(1);
180  }
181  // The delta phi angle between 2 adjacent slats is just 360/nslats. If
182  // there is just one crossing the tips of each slat can extend from
183  // phi-delta-phi/2 to phi+delta-phi/2. G4 rotates around the center of a
184  // slat so we are dealing in increments of just delta-phi/2. This
185  // has to be multiplied by the number of crossings we want.
186  //
187  // To find this tilt angle we have to calculate a triangle
188  // the long sides are the outer radius and the
189  // inner radius + 1/2 tracker thickness
190  // the angle between these sides is just (360/nslat)/2*(cross)
191  // the we use a/sin(alpha) = b/sin(beta)
192  // and c^2 = a^2+b^2 -2ab*cos(gamma)
193  // the solution is not unique, we have to pick 180-beta
194  // but the slat angle is 180-beta (it's on the other side of the triangle
195  // just draw the damned thing if this is confusing)
196  double sign = 1;
197  if (ncross < 0)
198  {
199  sign = -1;
200  }
201  int ncr = fabs(ncross);
202  double thick = TrackerThickness;
203  double c = radius + thick / 2.;
204  double b = radius + thick;
205 
206  double alpha = 0;
207  alpha = ((360. / _sciNum * M_PI / 180.)/2.) * ncr;
208  double sinb = sin(alpha) * b / (sqrt(b * b + c * c - 2 * b * c * cos(alpha)));
209  double beta = asin(sinb) * 180. / M_PI; // this is already the slat angle
210  _sciTilt = beta * sign;
211  // print out triangle
212 // double tbeta = 180 - beta;
213 // double gamma = 180 - (alpha * 180. / M_PI) - tbeta;
214 // double a = b * sin(alpha) / sinb;
215  // cout << "triangle length: a: " << a
216  // << ", b: " << b << ", c: " << c << endl;
217  // cout << "triangle angle: alpha: " << alpha*180./M_PI
218 // << ", beta: " << tbeta
219 // << ", gamma: " << gamma
220 // << endl;
221  cout << Name() << ": SetTiltViaNcross(" << ncross << ") setting slat angle to: " << _sciTilt << " degrees" << endl;
222  return;
223 }
